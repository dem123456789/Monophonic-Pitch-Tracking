1function [Pitch, Merit] = cmp_rate(Phi, Fs, maxcands, lag_min, lag_max,Prm)
  2% CMP_RATE compute rate and merit from the autocorrelation sequence
  3%
  4% [Pitch, Merit] = cmp_rate(Phi,Fs,maxcands,lag_min,lag_max,Prm);
  5%  This routine computes Pitch estimates and the corresponding Merit values 
  6%  associated with the peaks found in each frame based on the correlation sequence.
  7%
  8%  INPUTS:
  9%   Phi     : The input signal(either auto correlated/normalized cross correlated).
 10%   Fs      : The sampling frequency.
 11%   lag_min : The lowest lag(== 1/F0_max) involved in the F0 estimation.
 12%   F0_max  : The highest lag(== 1/F0_min)  involved in the F0 estimation.
 13%             greater than the peak at F0 during the first pass of search.
 14%   maxcands: The maximum number of top candidates to be considered.
 15%
 16%  OUTPUTS:
 17%    Pitch  : The Rate/Pitch values for the peaks found for each frame.
 18%    Merit  : The Merit values of the peaks found in each frame.
 19
 20%  Creation date:   2002
 21%  Revision dates:   March 26, 2002, December 24, 2005
 22%  Programmer: S.A.Zahorian,Kasi Kavita
 23
 24%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 25%     This file is a part of the YAAPT program, designed for a fundamental 
 26%   frequency tracking algorithm that is extermely robust for both high quality 
 27%   and telephone speech.  
 28%     The YAAPT program was created by the Speech Communication Laboratory of
 29%   the state university of New York at Binghamton. The program is available 
 30%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 31%   information about the program could be found at "A spectral/temporal 
 32%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 33%   June 2008.
 34%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 35
 36%-- PARAMETERS: set up all of these parameters --------------------------------
 37%Width of the window used in first pass of peak picking.
 38width = Prm.nccf_pwidth;
 39% The center of the window is defined as the peak location.
 40center = round(width/2); 
 41
 42
 43% The threshold value used for the first pass of
 44% peak picking for each frame.any peaks found greater than this
 45% are chosen in the first pass.
 46Merit_thresh1 = Prm.nccf_thresh1;
 47
 48%  The threshold value used to limit peak searching.
 49%  If a peak is found at least this great, no further  searching is
 50%  done increased from prev. value of 0.85.
 51Merit_thresh2 = Prm.nccf_thresh2;
 52
 53%-- INITIALIZATION -----------------------------------------------------------
 54numpeaks  = 0;
 55Pitch     = zeros(1, maxcands);
 56Merit     = zeros(1, maxcands);
 57
 58
 59%-- MAIN ROUTINE --------------------------------------------------------------
 60%  Find all peaks for a (lag_min--to--lag_max) search range
 61%   a "peak" must be the higher than a specfied number of
 62%   points on either side of point.  Peaks are later "cleaned"
 63%   up, to retain only best peaks i.e. peaks which do not meet certain
 64%   criteria are eliminated.
 65%   and allowing only the peaks which are a certain amplitude over the rest.
 66
 67for n = lag_min-center:lag_max
 68    [y, lag]  = max(Phi(n:n+width-1));
 69    if (lag == center) && (y > Merit_thresh1)
 70        
 71        numpeaks = numpeaks + 1;
 72        Pitch(numpeaks) = Fs/(n+lag-1);
 73        Merit(numpeaks) = y;
 74        
 75        if (y > Merit_thresh2)
 76            break;
 77        end;
 78    end;
 79end;
 80
 81% consider the case when the number of peaks are more than the maxcands.
 82% Then take only the best maxcands peaks based on the Merit values .
 83[Merit,Idx]=sort(Merit, 'descend');
 84Pitch = Pitch(Idx);
 85numpeaks = min(numpeaks, maxcands);
 86Merit = Merit(1:numpeaks);
 87Pitch  = Pitch(1:numpeaks);
 88
 89% if the number of peaks in the frame are less than the maxcands, then we 
 90% assign "null" values to remainder of peak and merit values in arrays
 91if (numpeaks < maxcands)
 92    if(isempty(Pitch))
 93        Pitch=zeros(1,maxcands);
 94        Merit=0.001*ones(1,maxcands);
 95    else
 96    Pitch = [Pitch zeros(1,maxcands-numpeaks)];
 97    Merit = [Merit 0.001*ones(1,maxcands-numpeaks)];
 98%     Pitch(numpeaks+1:maxcands) = 0;
 99%     Merit(numpeaks+1:maxcands) = 0.001;
100    end
101end;
102
103% Normlize merits
104Max_Merit = max(Merit);
105if (Max_Merit > 1.0)
106    Merit = Merit/Max_Merit;
107end
