1function [cnt] = fq2cnt(fq)
 2% FQ2CNT Transform input frequency to cent unit
 3%
 4% fq = input frequency
 5% cnt = output cent scale with base at 13.75 Hz
 6% Each piano key frequency is linearly separated 100 cents 
 7% Ex: A4 - Ab4
 8% fq2cnt(440) - fq2cnt(415.305) = 6000 - 5900 = 100
 9%
10cnt = 1200.*log2((fq./13.75));
11end
