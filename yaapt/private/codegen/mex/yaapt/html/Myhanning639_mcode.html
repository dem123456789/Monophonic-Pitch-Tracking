44% Evaluate the window
45% switch sflag,

52if ~rem(n,2)
53   % Even length window
54   half = n/2;
55   w = .5*(1 - cos(2*pi*(1:half)'/(n+1)));
56   w = [w; w(end:-1:1)];
57else
58   % Odd length window
59   half = (n+1)/2;
60   w = .5*(1 - cos(2*pi*(1:half)'/(n+1)));
61   w = [w; w(end-1:-1:1)];
62end

64end
