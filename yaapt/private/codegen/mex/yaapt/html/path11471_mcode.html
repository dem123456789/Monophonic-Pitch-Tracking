1function [PATH]=path1(Local,Trans)
  2%PATH1 find optimal path with the lowest cost
  3%
  4%This routine is used to find the optimal path with lowest cost if two matrice(Local cost matrix and Transition cost)
  5%are given.
  6%
  7% INPUTS: 
  8%   Local is the two dimentional cost matrix denating the local cost;
  9%   Trans is the three dimentional cost matrix denating the transition cost;
 10%
 11% OUTPUTS:
 12%   PATH is the lowest cost path of the given matrix;
 13
 14%   Programmers: Dr. Zahorian, Lingyun Gu
 15
 16%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 17%     This file is a part of the YAAPT program, designed for a fundamental 
 18%   frequency tracking algorithm that is extermely robust for both high quality 
 19%   and telephone speech.  
 20%     The YAAPT program was created by the Speech Communication Laboratory of
 21%   the state university of New York at Binghamton. The program is available 
 22%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 23%   information about the program could be found at "A spectral/temporal 
 24%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 25%   June 2008.
 26%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 27
 28[M,N]=size(Local);                              %getting the size of the local matrix;
 29
 30if M>=500
 31    error('Stop in Dynamic due to M>100')         %if M is greater than 100, stop the routine;
 32end
 33if N>=5000                                      %if N is greater than 1000, stop the routine;
 34    error('Stop in Dynamic due to N>1000')
 35end
 36
 37
 38PRED=ones(M,N);                                 %initializing several desired matrix;
 39P=ones(1,N);
 40p=zeros(1,N);
 41%PATH=zeros(1,N);
 42PCOST = zeros(1,M);
 43CCOST = zeros(1,M);
 44
 45for J=1:M                                     %initializing the previous costs;
 46    PCOST(J)=Local(J,1);     
 47end
 48
 49for I=2:N                                       %this loop is doing the heart work of this routine. That means to
 50    for J=1:M
 51        CCOST(J)=1.0E+30;                        %get the lowest cost path;
 52        
 53        for K=1:M
 54            if PCOST(K)+Trans(K,J,I)<=CCOST(J)  %deciding the optimal path between two points in two next column; 
 55                CCOST(J)=PCOST(K)+Trans(K,J,I);
 56                PRED(J,I)=K;                     %this line is very importent, used to mark the chosen points;
 57            end
 58        end
 59        
 60        if CCOST(J)>=1.0E+30
 61            error('CCOST>1.0E+50, Stop in Dynamic')
 62        end
 63        
 64        CCOST(J)=CCOST(J)+Local(J,I);            %new cost is gotten by the adding of Local cost and current cost;
 65        
 66    end
 67    
 68    for J=1:M
 69        PCOST(J)=CCOST(J);                      %using the new current cost to update the previous cost;
 70    end
 71    
 72    p(I)=1;
 73    for J=2:M                                    %obtaining the points with lowest cost in every column;
 74        if CCOST(J)<=CCOST(1)
 75            CCOST(1)=CCOST(J);p(I)=J;
 76        end
 77    end
 78end
 79
 80%  Determine ending state with lowest cost
 81%  Note that CCOST array is filled with costs of states for ending time step
 82%  at this point in code
 83
 84%  IT would seem that following lines of code should be able
 85%  to be substituted for the above similar lines, which are used every iteration
 86%  However, results are same with clean speech, but slightly degraded with noisy
 87%  speech
 88
 89%   p(N) = 1;
 90%   for J=2:M                   %obtaining the points with lowest cost in every column;
 91%     if CCOST(J)<=CCOST(1)
 92%          CCOST(1)=CCOST(J);p(N)=J;
 93%     end
 94%   end
 95
 96
 97P(N)=p(N);
 98
 99for I=N-1:-1:1                   %using this loop to get the path finally; from the last point going
100   P(I)=PRED(P(I+1),I+1);        %backward to find the previous points, etc;
101end
102PATH=P;                          %getting the final path. 
103
104
