1function[SPitch, VUVSPitch, pAvg, pStd]= spec_trk(Data, Fs, VUVEnergy, Prm)
  2%SPEC_TRK  Spectral pitch tracking for YAAPT pitch tracking
  3%
  4%   [SPitch, VUVSPitch, pAvg, pStd]=spec_trk(Data, Fs, VUVEnergy, Prm) 
  5%   computes estimates of pitch  using nonlinearly processed
  6%   speech (typically square or absolute value) and frequency domain processing
  7%   Search for frequencies which have energy at multiplies of that frequency
  8%
  9%INPUTS:
 10%   Data:      Nonlinearly processed signal, filtered with F1. (~50 to 1500Hz)
 11%   Fs:        The sampling frequency.
 12%   VUVEnergy: Voiced/ unvoiced decision, depending on nlfer
 13%   Prm:       Parameters
 14%
 15%OUTPUTS:
 16%   SPitch:    The spectral Pitch track, with the unvoiced regions filled using interpolation.
 17%   VUVSPitch: The spectral Pitch track, with  unvoiced regions set at zero
 18%   pAvg:      The average of the Pitch track value.
 19%   pStd:      The standard deviation in the track.
 20
 21%   Creation date:  Feb 20, 2006
 22%   Revision dates: Feb 22, 2006,  March 11, 2006, April 5, 2006,
 23%                   Jun 25, 2006,  June 13, 2007
 24%   Programers:     Hongbing Hu, Princy, Zahorian
 25
 26%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 27%     This file is a part of the YAAPT program, designed for a fundamental 
 28%   frequency tracking algorithm that is extermely robust for both high quality 
 29%   and telephone speech.  
 30%     The YAAPT program was created by the Speech Communication Laboratory of
 31%   the state university of New York at Binghamton. The program is available 
 32%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 33%   information about the program could be found at "A spectral/temporal 
 34%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 35%   June 2008.
 36%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 37
 38%-- PARAMETERS: set up all of these parameters --------------------------------
 39nframesize = fix(Prm.frame_length*Fs/1000);    
 40nframejump = fix(Prm.frame_space*Fs/1000); 
 41noverlap= nframesize-nframejump;         % overlap width in sample 
 42numframes = fix((length(Data)-noverlap)/nframejump);
 43
 44% use larger frame size
 45nframesize = nframesize*2;
 46
 47% Max number of peak candidates found
 48maxpeaks = Prm.shc_maxpeaks;
 49nfftlength = Prm.fft_length;                       % FFT length
 50% Resolution of spectrum
 51delta         = Fs/nfftlength;               
 52% Window width in sample
 53window_length = fix(Prm.shc_window/delta);
 54if (mod(window_length,2)==0)
 55    window_length = window_length + 1;
 56end
 57% Half window width
 58half_winlen   = fix(Prm.shc_window/(delta*2)); 
 59% Max range of SHC 
 60max_SHC   = fix((Prm.f0_max+Prm.shc_pwidth*2)/delta);
 61% Min range of SHC 
 62min_SHC   = ceil(Prm.f0_min/delta);       
 63% Number of harmomics considered
 64numharmonics = Prm.shc_numharms;
 65
 66
 67%-- INITIALIZATION -----------------------------------------------------------
 68CandsPitch = zeros(maxpeaks, numframes);
 69CandsMerit = ones(maxpeaks, numframes);
 70% Zero paddinga
 71% a = Data;
 72% Data(end:(numframes-1)*nframejump+nframesize) = 0;
 73Data(end) = 0;
 74Data = [Data zeros(1,(numframes-1)*nframejump+nframesize-length(Data))];
 75%-- MAIN ROUTINE --------------------------------------------------------------
 76% Compute SHC for voiced frame
 77Kaiser_window = Mykaiser(nframesize);
 78% Kaiser_window_test = kaiser(nframesize);
 79SHC = zeros(1,max_SHC);
 80
 81winix = repmat([1:window_length], numharmonics+1,1);
 82rowix = repmat([1:numharmonics+1]', 1, window_length);
 83
 84for frame = 1:numframes
 85    if (VUVEnergy(frame) > 0)
 86        firstp = 1+(frame-1)*(nframejump);
 87        Signal = Data(firstp:firstp+nframesize-1) .* Kaiser_window';
 88        Signal = Signal - mean(Signal);
 89%            Magnit = [zeros(half_winlen, 1); abs(fft(Signal , nfftlength))];
 90             Magnit = [zeros(half_winlen, 1); abs(fft(Signal , nfftlength))'];
 91
 92        % Compute SHC (Spectral Harmonic Correlation)
 93        for k=min_SHC:max_SHC;
 94            SHC(k) = sum(prod(Magnit( winix + k*rowix )));
 95        end
 96        [CandsPitch(:,frame), CandsMerit(:,frame)]=peaks(SHC,delta, maxpeaks, Prm);
 97    else 
 98        % if energy is low,  let frame be considered as unvoiced
 99        CandsPitch(:,frame) = zeros(1,maxpeaks);
100        CandsMerit(:,frame) = ones(1, maxpeaks);
101    end
102end
103
104% display('Running new code');
105
106% for frame = 1:numframes
107%     if (VUVEnergy(frame) > 0)
108%         firstp = 1+(frame-1)*(nframejump);
109%         Signal = Data(firstp:firstp+nframesize-1) .* Kaiser_window;
110%         Signal = Signal - mean(Signal);
111%         Magnit = abs(fft(Signal , nfftlength));
112%         
113%         % Compute SHC (Spectral Harmonic Correlation)
114%         for k=min_SHC:max_SHC;
115%             MultHarms = ones(1, window_length);
116% 
117%             % Set each harmonics, 1=f0 2=2*f0 etc
118%             for n=1:numharmonics+1  % Number of harmomics considered
119%                 nstart = n*k-half_winlen;
120%                 if nstart < 0
121%                     Harm =  zeros(1, window_length);
122%                     Harm(abs(nstart)+1:window_length)=Magnit(1:nstart+window_length);
123%                 else
124%                     Harm(1:window_length)=Magnit(nstart+1:nstart+window_length);
125%                 end
126%                 MultHarms = MultHarms .* Harm;
127%             end
128%             SHC(k) = sum(MultHarms);
129%         end
130%         [CandsPitch(:,frame), CandsMerit(:,frame)]=peaks(SHC,delta, maxpeaks, Prm);
131%     else 
132%         % if energy is low,  let frame be considered as unvoiced
133%         CandsPitch(:,frame) = zeros(1,maxpeaks);
134%         CandsMerit(:,frame) = ones(1, maxpeaks);
135%     end
136% end
137
138% Extract the Pitch candidates of voiced frames for the future Pitch selection 
139SPitch = CandsPitch(1,:);
140Idx_voiced = SPitch > 0; 
141VCandsPitch  = CandsPitch(:,Idx_voiced);
142VCandsMerit = CandsMerit(:,Idx_voiced);
143Num_VCands = sum(Idx_voiced);
144
145% Average, STD of the first choice candidates
146avg_voiced = mean(VCandsPitch(1,:));
147std_voiced = std(VCandsPitch(1,:));
148
149% Weight the deltas, so that higher merit candidates are considered
150% more favorably
151delta1 = abs((VCandsPitch - 0.8*avg_voiced)).*(3-VCandsMerit);
152
153% Interpolation of the weigthed candidates
154[VR, Idx] = min(delta1); 
155VPeak_minmrt  = zeros(1, length(Idx));
156VMerit_minmrt = zeros(1, length(Idx));
157for n=1: length(Idx)
158    VPeak_minmrt(n)  = VCandsPitch(Idx(n), n);
159    VMerit_minmrt(n) = VCandsMerit(Idx(n), n); 
160end
161VPeak_minmrt = Mymedfilt1(VPeak_minmrt, max(1,Prm.median_value-2));
162% VPeak_minmrt_test = medfilt1(VPeak_minmrt, max(1,Prm.median_value-2));
163% Replace the lowest merit candidates by the median smoothed ones
164% computed from highest merit peaks above
165for n=1: length(Idx)
166    VCandsPitch(Idx(n), n) = VPeak_minmrt(n);
167    % Assign merit for the smoothed peaks
168    VCandsMerit(Idx(n), n) = VMerit_minmrt(n);
169end
170
171% Use dynamic programming to find best overal path among pitch candidates
172% Dynamic weight for transition costs
173% balance between local and transition costs
174weight_trans = Prm.dp5_k1*std_voiced/avg_voiced;
175VPitch = [];
176if (Num_VCands >2) 
177  [VPitch] = dynamic5(VCandsPitch, VCandsMerit,weight_trans);
178   VPitch = Mymedfilt1(VPitch, max(Prm.median_value-2, 1));
179%    VPitch_test = medfilt1(VPitch, max(Prm.median_value-2, 1));
180   
181else
182     
183    if (Num_VCands > 0) 
184%       for i = 1:Num_VCands   % assume at least 1 voiced candidate
185%        VPitch (i) = 150;   %   Not really correct,  but should prevent hang up if only  0,1,2,3 voiced candidates
186%       end; 
187        VPitch(1:Num_VCands) = 150;
188                %   This should only occur for very short pitch tracks, and even then rarely
189    else
190        VPitch(1) = 150;    % to handle when no real voiced candidates
191        Idx_voiced(1) = 1;
192    end;
193    
194end;       
195
196
197% Computing some statistics from the voiced frames
198pAvg = mean(VPitch,2);
199pStd = std(VPitch,0,2);
200
201% Streching out the smoothed pitch track
202SPitch(Idx_voiced) = VPitch;
203
204% Interpolating thru unvoiced frames
205SPitch_temp_first = SPitch(1);
206SPitch_temp_end = SPitch(end);
207if all(SPitch_temp_first < (pAvg/2))
208    SPitch(1)   = pAvg;   
209end
210if all(SPitch_temp_end < (pAvg/2))
211    SPitch(end) = pAvg;  
212end
213
214[Indrows, Indcols, Vals] = find(SPitch);
215SPitch = interp1(Indcols, Vals, [1:numframes], 'pchip');
216
217FILTER_ORDER = 3;
218flt_b = (ones(1,FILTER_ORDER))/FILTER_ORDER ;
219flt_a = 1;
220
221SPitch = filter(flt_b, flt_a, SPitch);
222
223
224%  above messes up  first few values of SPitch  ---  simple fix up
225%  Note--   this fix up should be based on above filter order
226
227SPitch(1) = SPitch(3);
228SPitch(2) = SPitch (4);
229
230% Create pitch track with Voiced/Unvoiced decision
231VUVSPitch = SPitch;
232VUVSPitch(VUVEnergy==0) = 0;
233
234% SPitch = SPitch - delta;
