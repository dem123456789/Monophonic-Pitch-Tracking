24function m = Mymedfilt1(s, w)
25        if nargin == 1,
26                w = 5;
27        end
28        
29        s = s(:)';
30        w2 = floor(w/2);
31        w = 2*w2 + 1;
32
33        n = length(s);
34        m = zeros(w,n+w-1);
35        s0 = s(1); sl = s(n);
36
37        for i=0:(w-1),
38                m(i+1,:) = [s0*ones(1,i) s sl*ones(1,w-i-1)];
39        end
40        m = median(m);
41        m = m(w2+1:w2+n);
42end
