  514           i38 = Pitch->size[1];

  528               i39 = Trans->size[2];

  538           loop_ub++;
  539           emlrtBreakCheckFastR2012b(emlrtBreakCheckR2012bFlagVar, sp);
  540         }
  541   
  542         b_Pitch++;
  543         emlrtBreakCheckFastR2012b(emlrtBreakCheckR2012bFlagVar, sp);
  544       }

  546       i++;
  547       emlrtBreakCheckFastR2012b(emlrtBreakCheckR2012bFlagVar, sp);

  552     i37 = Trans->size[0] * Trans->size[1] * Trans->size[2];
  553     emxEnsureCapacity(sp, (emxArray__common *)Trans, i37, (int32_T)sizeof(real_T),

  559     for (i37 = 0; i37 < loop_ub; i37++) {
