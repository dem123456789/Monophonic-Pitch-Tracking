1function [FinPitch] = dynamic(Pitch, Merit, Energy, Prm)
  2%DYNAMIC Dynamic programming for YAAPT pitch tracking
  3%
  4% [FinPitch] = dynamic(Pitch, Merit, Energy, Prm) is used
  5% to compute Local and Transition cost matrices
  6% to enable lowest cost tracking of pitch candidates;
  7% It uses NFLER from the spectrogram and the highly robust
  8% spectral F0 track, plus the merits, for
  9% computation of the cost matrices.
 10%
 11%INPUTS: 
 12%   Pitch:  Pitch array with elements arranged so that lowest
 13%                      index correponds to  highest merit
 14%   Merit:  Merit array with elements arranged as for pitch
 15%   Energy: NLFER energy ratio from the low frequency regions of spectrogram.
 16%   Prm:    Array of parameters used to control performance
 17%
 18%OUTPUTS:  
 19%   FinPitch: The final pitch track for the entire utterance pitch
 20%   values are in HZ.Unvoiced frames are assigned 0.
 21
 22% Note:   This routine is intended for more through testing of
 23%    some thresholds used in dynamic6.   All the dp constants are
 24%    set to fixed values that worked well in testing
 25%    then the dp constants are used to specify the thresholds and other
 26%     constants
 27
 28%   Creation date:   Spring 2001
 29%   Revision date:   January 3, 2002   June 16, 2007, July 27, 2007
 30%   Programmers: S. Zahorian,Kavita Kasi, Lingyun Gu, Hongbing Hu
 31
 32%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 33%     This file is a part of the YAAPT program, designed for a fundamental 
 34%   frequency tracking algorithm that is extermely robust for both high quality 
 35%   and telephone speech.  
 36%     The YAAPT program was created by the Speech Communication Laboratory of
 37%   the state university of New York at Binghamton. The program is available 
 38%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 39%   information about the program could be found at "A spectral/temporal 
 40%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 41%   June 2008.
 42%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 43
 44
 45%-- INITIALIZATION -----------------------------------------------------------
 46[numcands, numframes] = size(Pitch);
 47
 48%Copy some arrays
 49BestPitch  = Pitch(numcands-1,:);
 50mean_pitch = mean(BestPitch(BestPitch>0));
 51
 52%The following weighting factors are used to differentially weight
 53% the various types of transitions which can occur, as well as weight
 54% the relative value of transition costs and local costs
 55dp_w1 = Prm.dp_w1;
 56dp_w2 = Prm.dp_w2;
 57dp_w3 = Prm.dp_w3;
 58dp_w4 = Prm.dp_w4;
 59
 60
 61%-- MAIN ROUTINE --------------------------------------------------------------
 62% Forming the local cost matrix
 63Local = 1 - Merit;
 64
 65% Initialization for the formation of the transition cost matrix
 66Trans  = ones(numcands,numcands,numframes);
 67
 68% The transition cost matrix is proportional to frequency differences
 69% between successive candidates.
 70for i = 2:numframes
 71    for j = 1:numcands
 72        for k = 1:numcands
 73            % both candidates voiced
 74            if ((Pitch(j,i) > 0) && (Pitch(k,i-1) > 0) )        
 75                Trans(k,j,i) = dp_w1*(abs(Pitch(j,i)-Pitch(k,i-1))/mean_pitch);
 76            end
 77            
 78            % one candiate is unvoiced
 79            if (Pitch(j,i)==0 && Pitch(k,i-1)>0) || (Pitch(j,i)>0 && Pitch(k,i-1)==0)
 80                benefit = min(1, abs(Energy(i-1)-Energy(i)));
 81                Trans(k,j,i) =  dp_w2*(1-benefit);
 82            end
 83            
 84            % both candidates are unvoiced
 85            if ((Pitch(j,i) == 0) && (Pitch(k,i-1) == 0))
 86                Trans(k,j,i) =  dp_w3;
 87            end
 88        end
 89    end
 90end
 91
 92% Overal balance between Local and Transition costs
 93Trans = Trans/dp_w4;
 94
 95% Find the minimum cost path thru Pitch_Array using the Local and Trans costs
 96Path = path1(Local,Trans);
 97
 98%extracting the pitch, using Path
 99FinPitch = zeros(1,numframes);
100for i = 1:numframes
101  FinPitch(i) = Pitch(Path(i),i);
102end
103
