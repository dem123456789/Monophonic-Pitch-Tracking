1function[Energy, VUVEnergy]= nlfer(Data, Fs, Prm)
 2% NLFER  Normalized Low Frequency Energy Ratio
 3%
 4%   [Energy, VUVEnergy]= nlfer(Data, Fs, Prm) computes the
 5%   nromlized low frequency energy ratio.
 6%
 7% INPUTS:
 8%   Data:  Nonlinear, filtered signal for NLFER caculation
 9%   Fs:    The sampling frequency.
10%   Prm:   Array of parameters used to control performance
11%
12% OUTPUTS:
13%   Energy:     NLFER Energy of the data
14%   VUVEnergy:  Voiced/Unvoiced decisions (optimum??)
15
16%   Creation date:  Oct 17, 2006, July 13, 2007
17%   Programers:     Hongbing Hu, S. Zahorian
18
19%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
20%     This file is a part of the YAAPT program, designed for a fundamental 
21%   frequency tracking algorithm that is extermely robust for both high quality 
22%   and telephone speech.  
23%     The YAAPT program was created by the Speech Communication Laboratory of
24%   the state university of New York at Binghamton. The program is available 
25%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
26%   information about the program could be found at "A spectral/temporal 
27%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
28%   June 2008.
29%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
30
31%-- PARAMETERS: set up all of these parameters --------------------------------
32nfftlength = Prm.fft_length;            % FFT length
33nframesize = fix(Prm.frame_length*Fs/1000);    
34nframejump = fix(Prm.frame_space*Fs/1000); 
35
36% If normalized low-frequency energy is below this, assume unvoiced frame
37nlfer_thersh1  = Prm.nlfer_thresh1;
38
39% Low freqeuncy range for NLFER
40N_F0_min = round ((Prm.f0_min*2/Fs) * nfftlength );
41N_F0_max = round ((Prm.f0_max/Fs) * nfftlength );
42
43%-- MAIN ROUTINE --------------------------------------------------------------
44
45%  Spectrogram of the data
46SpecData = Myspecgram(Data,nfftlength,Fs,(nframesize),(nframesize-nframejump));
47
48% Compute normalize low-frequency energy ratio 
49FrmEnergy = sum(abs(SpecData(N_F0_min:N_F0_max,:)));
50avgEnergy = mean(FrmEnergy);
51
52Energy = FrmEnergy/avgEnergy;
53% The frame is voiced if NLFER enery > threshold, otherwise is unvoiced.
54VUVEnergy = (Energy > nlfer_thersh1);
55
