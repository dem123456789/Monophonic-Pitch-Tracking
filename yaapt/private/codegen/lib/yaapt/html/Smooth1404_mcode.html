1function [out] = Smooth(arr)
  2% Smooth frequency data by 3 chunks
  3% arr = frequency data
  4% out = data after smoothed
  5% 3 chunks is one window, if the lengh of the middle chunk is below
  6% threshold it will be absorbed by the head or tail chunk.
  7chunk1_value = 0;
  8chunk2_value = 0;
  9chunk3_value = 0;
 10chunk1 = [];
 11chunk2 = [];
 12chunk3 = [];
 13pointer = 0;
 14fixed_threshold = 10; % 10 frames
 15percent_threshold = 0.01; % 1 percent
 16for i=1:length(arr)
 17    if pointer== 0 
 18       chunk1_value = arr(i);
 19       chunk1 = i;
 20       pointer = pointer + 1;
 21    elseif pointer == 1     
 22        if isempty(chunk1)
 23            chunk1_value=arr(i);
 24            chunk1 = i;
 25        else
 26            if arr(i)==chunk1_value
 27                chunk1 = [chunk1 i];
 28            else
 29                pointer = pointer +1;
 30                chunk2_value = arr(i);
 31                chunk2 = i;
 32            end
 33        end
 34    elseif pointer == 2
 35        if arr(i)==chunk2_value
 36            chunk2 = [chunk2 i];
 37        else
 38            pointer = pointer +1;
 39            chunk3_value = arr(i);
 40            chunk3 = i;
 41        end
 42    elseif pointer == 3
 43        if arr(i)==chunk3_value
 44            chunk3 = [chunk3 i];
 45        else
 46            % new chunk found
 47            len_tmp=[length(chunk1), length(chunk2), length(chunk3)];
 48            [min_len,idx] = min(len_tmp);
 49            if(min_len<=fixed_threshold)
 50                if idx == 1
 51                    for m=1:length(chunk1)
 52                        arr(chunk1(m)) = chunk2_value;
 53                    end
 54                elseif idx == 2
 55                    len_tmp(idx) = max(len_tmp)+100;
 56                    [~,second_min_idx] = min(len_tmp);
 57                    if second_min_idx == 1
 58                        for m=1:length(chunk2)
 59                            arr(chunk2(m)) = chunk1_value;
 60                        end
 61                    elseif second_min_idx == 3
 62                        for m=1:length(chunk2)
 63                            arr(chunk2(m)) = chunk3_value;
 64                        end
 65                    end
 66                elseif idx == 3
 67                    for m=1:length(chunk3)
 68                        arr(chunk3(m)) = chunk2_value;
 69                    end
 70                end
 71            end
 72            chunk1_value = arr(i);
 73            chunk2_value = 0;
 74            chunk3_value = 0;
 75            chunk1 = i;
 76            chunk2 = [];
 77            chunk3 = [];
 78            pointer = 1;
 79        end
 80    end
 81    if i==length(arr)
 82        if pointer == 2
 83            if length(chunk1)<fixed_threshold
 84                for m=1:length(chunk1)
 85                    arr(chunk1(m)) = chunk2_value;
 86                end     
 87            end
 88        elseif pointer == 3
 89            len_tmp=[length(chunk1), length(chunk2), length(chunk3)];
 90            [min_len,idx] = min(len_tmp);
 91            if(min_len<=fixed_threshold)
 92                if idx == 1
 93                    for m=1:length(chunk1)
 94                        arr(chunk1(m)) = chunk2_value;
 95                    end
 96                elseif idx == 2
 97                    len_tmp(idx) = max(len_tmp)+100;
 98                    [~,second_min_idx] = min(len_tmp);
 99                    if second_min_idx == 1
100                        for m=1:length(chunk2)
101                            arr(chunk2(m)) = chunk1_value;
102                        end
103                    elseif second_min_idx == 3
104                        for m=1:length(chunk2)
105                            arr(chunk2(m)) = chunk3_value;
106                        end
107                    end
108                elseif idx == 3
109                    for m=1:length(chunk3)
110                        arr(chunk3(m)) = chunk2_value;
111                    end
112                end
113            end
114        end
115    end
116end
117out = arr;
118end
