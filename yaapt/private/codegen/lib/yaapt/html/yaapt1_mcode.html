  3% YAAPT Fundamental Frequency (Pitch) tracking
  4%
  5% [Pitch, numfrms, frmrate] = yaapt(Data, Fs, VU, ExtrPrm, fig)
  6%   , the function is to check input parameters and invoke a number of associated routines 
  7%   for the YAAPT pitch tracking.
  8%
  9% INPUTS: 
 10%   Data:       Input speech raw data
 11%   Fs:         Sampling rate of the input data
 12%   VU:         Whether to use voiced/unvoiced decision with 1 for True and 0 for 
 13%               False.The default is 1.
 14%   ExtrPrm:    Extra parameters in a struct type for performance control.
 15%               See available parameters defined in yaapt.m 
 16%               e.g., 
 17%               ExtrPrm.f0_min = 60;         % Change minimum search F0 to 60Hz 
 18%               ExtrmPrm.fft_length = 8192;  % Change FFT length to 8192
 19%   fig:        Whether to plot pitch tracks, spectrum, engergy, etc. The parameter
 20%               is 1 for True and 0 for False. The default is 0.   
 21%
 22% OUTPUTS:
 23%   Pitch:      Final pitch track in Hz. Unvoiced frames are assigned to 0s.
 24%   numfrms:    Total number of calculated frames, or the length of
 25%               output pitch track
 26%   frmrate:    Frame rate of output pitch track in ms
 27
 28%  Creation Date:  June 2000
 29%  Revision date:  Jan 2, 2002 , Jan 13, 2002 Feb 19, 2002, Mar 3, 2002
 30%                  June 11, 2002, Jun 30, 2006, July 27, 2007
 31%                  May 20, 2012: Add the VU parameter for whether to use
 32%                  voiced/unvoiced decision. 
 33%  Authors:        Hongbing Hu, Stephen A.Zahorian
 34
 35%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 36%     This file is a part of the YAAPT program, designed for a fundamental 
 37%   frequency tracking algorithm that is extermely robust for both high quality 
 38%   and telephone speech.  
 39%     The YAAPT program was created by the Speech Communication Laboratory of
 40%   the state university of New York at Binghamton. The program is available 
 41%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 42%   information about the program can be found at "A spectral/temporal 
 43%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 44%   June 2008.
 45%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 53% Default values for the tracking with voiced/unvoiced decision
 54Prm = struct(...
 55    'frame_length',   Parameter(1), ... % Length of each analysis frame (ms)
 56    'frame_space',    Parameter(2), ... % Spacing between analysis frame (ms)
 57    'f0_min',         Parameter(3), ... % Minimum F0 searched (Hz)
 58    'f0_max',        Parameter(4), ... % Maximum F0 searached (Hz)
 59    'fft_length',   Parameter(5), ... % FFT length
 60    'bp_forder',     Parameter(6), ... % Order of bandpass filter (even)
 61    'bp_low',         Parameter(7), ... % Low frequency of filter passband (Hz)
 62    'bp_high',      Parameter(8), ... % High frequency of filter passband (Hz)
 63    'nlfer_thresh1',Parameter(9), ... % NLFER boundary for voiced/unvoiced decisions
 64    'nlfer_thresh2', Parameter(10), ... % Threshold for NLFER definitely unvocied
 65    'shc_numharms',    Parameter(11), ... % Number of harmonics in SHC calculation
 66    'shc_window',     Parameter(12), ... % SHC window length (Hz)
 67    'shc_maxpeaks',    Parameter(13), ... % Maximum number of SHC peaks to be found
 68    'shc_pwidth',     Parameter(14), ... % Window width in SHC peak picking (Hz)
 69    'shc_thresh1',   Parameter(15), ... % Threshold 1 for SHC peak picking
 70    'shc_thresh2',  Parameter(16), ... % Threshold 2 for SHC peak picking
 71    'f0_double',     Parameter(17), ... % F0 doubling decision threshold (Hz)
 72    'f0_half',       Parameter(18), ... % F0 halving decision threshold (Hz)
 73    'dp5_k1',         Parameter(19), ... % Weight used in dynaimc program
 74    'dec_factor',      Parameter(20), ... % Factor for signal resampling
 75    'nccf_thresh1', Parameter(21), ... % Threshold for considering a peak in NCCF
 76    'nccf_thresh2',  Parameter(22), ... % Threshold for terminating serach in NCCF
 77    'nccf_maxcands',   Parameter(23), ... % Maximum number of candidates found
 78    'nccf_pwidth',     Parameter(24), ... % Window width in NCCF peak picking
 79    'merit_boost',  Parameter(25), ... % Boost merit
 80    'merit_pivot',  Parameter(26), ... % Merit assigned to unvoiced candidates in
 81                          ... % defintely unvoiced frames
 82    'merit_extra',   Parameter(27), ... % Merit assigned to extra candidates
 83                          ... % in reducing F0 doubling/halving errors
 84    'median_value',    Parameter(28), ... % Order of medial filter
 85    'dp_w1',        Parameter(29), ... % DP weight factor for V-V transitions
 86    'dp_w2',         Parameter(30), ... % DP weight factor for V-UV or UV-V transitions
 87    'dp_w3',         Parameter(31), ... % DP weight factor of UV-UV transitions
 88    'dp_w4',         Parameter(32), ... % Weight factor for local costs
 89    'end', Parameter(33));

117%-- MAIN ROUTINE --------------------------------------------------------------
118%  Step 1. Preprocessing
119%  Create the squared or absolute values of filtered speech data
120Data_after = Preprocess(Data);
121[DataB, DataC, DataD, nFs] = nonlinear(Data_after, Fs, Prm);
122
123%  Check frame size, frame jump and the number of frames for nonlinear singal
124nframesize = fix(Prm.frame_length*nFs/1000);    
125if (nframesize < 15)
126    error('Frame length value %d is too short', Prm.frame_length);
127end
128if (nframesize > 2048)
129    error('Frame length value %d exceeds the limit', Prm.frame_length);
130end
131
132%  Step 2. Spectral pitch tracking
133%  Calculate NLFER and determine voiced/unvoiced frames with NLFER
134[Energy, VUVEnergy]= nlfer(DataB, nFs, Prm);
135
136%  Calculate an approximate pitch track from the spectrum.
137%  At this point, SPitch is best estimate of pitch track from spectrum
138[SPitch, VUVSPitch, pAvg, pStd]= spec_trk(DataD, nFs, VUVEnergy, Prm);
139
140
141%  Step 3. Temporal pitch tracking based on NCCF
142%  Calculate a pitch track based on time-domain processing
143%  Pitch tracking from the filterd original signal 
144[TPitch1, TMerit1] = tm_trk(DataB, nFs, SPitch, pStd, pAvg, Prm);
145
146%  Pitch tracking from the filterd nonlinear signal 
147[TPitch2, TMerit2] = tm_trk(DataD, nFs, SPitch, pStd, pAvg, Prm);
148
149% Refine pitch candidates 
150[RPitch, Merit] = refine(TPitch1, TMerit1, TPitch2, TMerit2, SPitch, ...
151                        Energy, VUVEnergy, Prm);
152
153% Step 5. Use dyanamic programming to determine the final pitch
154Pitch_before  = dynamic(RPitch, Merit, Energy, Prm);
155numfrms = length(Pitch_before);
156frmrate = Prm.frame_space; 
157 
158[Pitch_Freq,~] = freqSelect(Pitch_before);
159Pitch = Pitch_Optimization(Pitch_Freq);
