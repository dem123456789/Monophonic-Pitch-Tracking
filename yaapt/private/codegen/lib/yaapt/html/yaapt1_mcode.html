  3% YAAPT Fundamental Frequency (Pitch) tracking
  4%
  5% [Pitch, numfrms, frmrate] = yaapt(Data, Fs, VU, ExtrPrm, fig)
  6%   , the function is to check input parameters and invoke a number of associated routines 
  7%   for the YAAPT pitch tracking.
  8%
  9% INPUTS: 
 10%   Data:       Input speech raw data
 11%   Fs:         Sampling rate of the input data
 12%   VU:         Whether to use voiced/unvoiced decision with 1 for True and 0 for 
 13%               False.The default is 1.
 14%   ExtrPrm:    Extra parameters in a struct type for performance control.
 15%               See available parameters defined in yaapt.m 
 16%               e.g., 
 17%               ExtrPrm.f0_min = 60;         % Change minimum search F0 to 60Hz 
 18%               ExtrmPrm.fft_length = 8192;  % Change FFT length to 8192
 19%   fig:        Whether to plot pitch tracks, spectrum, engergy, etc. The parameter
 20%               is 1 for True and 0 for False. The default is 0.   
 21%
 22% OUTPUTS:
 23%   Pitch:      Final pitch track in Hz. Unvoiced frames are assigned to 0s.
 24%   numfrms:    Total number of calculated frames, or the length of
 25%               output pitch track
 26%   frmrate:    Frame rate of output pitch track in ms
 27
 28%  Creation Date:  June 2000
 29%  Revision date:  Jan 2, 2002 , Jan 13, 2002 Feb 19, 2002, Mar 3, 2002
 30%                  June 11, 2002, Jun 30, 2006, July 27, 2007
 31%                  May 20, 2012: Add the VU parameter for whether to use
 32%                  voiced/unvoiced decision. 
 33%  Authors:        Hongbing Hu, Stephen A.Zahorian
 34
 35%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 36%     This file is a part of the YAAPT program, designed for a fundamental 
 37%   frequency tracking algorithm that is extermely robust for both high quality 
 38%   and telephone speech.  
 39%     The YAAPT program was created by the Speech Communication Laboratory of
 40%   the state university of New York at Binghamton. The program is available 
 41%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 42%   information about the program can be found at "A spectral/temporal 
 43%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 44%   June 2008.
 45%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

 53% Default values for the tracking with voiced/unvoiced decision
 54Prm = struct(...
 55    'frame_length',   25, ... % Length of each analysis frame (ms)
 56    'frame_space',    10, ... % Spacing between analysis frame (ms)
 57    'f0_min',         60, ... % Minimum F0 searched (Hz)
 58    'f0_max',        400, ... % Maximum F0 searached (Hz)
 59    'fft_length',   8192, ... % FFT length
 60    'bp_forder',     150, ... % Order of bandpass filter
 61    'bp_low',         50, ... % Low frequency of filter passband (Hz)
 62    'bp_high',      1500, ... % High frequency of filter passband (Hz)
 63    'nlfer_thresh1',0.75, ... % NLFER boundary for voiced/unvoiced decisions
 64    'nlfer_thresh2', 0.1, ... % Threshold for NLFER definitely unvocied
 65    'shc_numharms',    3, ... % Number of harmonics in SHC calculation
 66    'shc_window',     40, ... % SHC window length (Hz)
 67    'shc_maxpeaks',    4, ... % Maximum number of SHC peaks to be found
 68    'shc_pwidth',     50, ... % Window width in SHC peak picking (Hz)
 69    'shc_thresh1',   5.0, ... % Threshold 1 for SHC peak picking
 70    'shc_thresh2',  1.25, ... % Threshold 2 for SHC peak picking
 71    'f0_double',     150, ... % F0 doubling decision threshold (Hz)
 72    'f0_half',       150, ... % F0 halving decision threshold (Hz)
 73    'dp5_k1',         11, ... % Weight used in dynaimc program
 74    'dec_factor',      1, ... % Factor for signal resampling
 75    'nccf_thresh1', 0.25, ... % Threshold for considering a peak in NCCF
 76    'nccf_thresh2',  0.9, ... % Threshold for terminating serach in NCCF
 77    'nccf_maxcands',   3, ... % Maximum number of candidates found
 78    'nccf_pwidth',     5, ... % Window width in NCCF peak picking
 79    'merit_boost',  0.20, ... % Boost merit
 80    'merit_pivot',  0.99, ... % Merit assigned to unvoiced candidates in
 81                          ... % defintely unvoiced frames
 82    'merit_extra',   0.4, ... % Merit assigned to extra candidates
 83                          ... % in reducing F0 doubling/halving errors
 84    'median_value',    7, ... % Order of medial filter
 85    'dp_w1',        0.15, ... % DP weight factor for V-V transitions
 86    'dp_w2',         0.5, ... % DP weight factor for V-UV or UV-V transitions
 87    'dp_w3',         0.1, ... % DP weight factor of UV-UV transitions
 88    'dp_w4',         0.9, ... % Weight factor for local costs
 89    'end', -1);

117%-- MAIN ROUTINE --------------------------------------------------------------
118%  Step 1. Preprocessing
119%  Create the squared or absolute values of filtered speech data
120[DataB, DataC, DataD, nFs] = nonlinear(Data, Fs, Prm);
121
122%  Check frame size, frame jump and the number of frames for nonlinear singal
123nframesize = fix(Prm.frame_length*nFs/1000);    
124if (nframesize < 15)
125    error('Frame length value %d is too short', Prm.frame_length);
126end
127if (nframesize > 2048)
128    error('Frame length value %d exceeds the limit', Prm.frame_length);
129end
130
131%  Step 2. Spectral pitch tracking
132%  Calculate NLFER and determine voiced/unvoiced frames with NLFER
133[Energy, VUVEnergy]= nlfer(DataB, nFs, Prm);
134
135%  Calculate an approximate pitch track from the spectrum.
136%  At this point, SPitch is best estimate of pitch track from spectrum
137[SPitch, VUVSPitch, pAvg, pStd]= spec_trk(DataD, nFs, VUVEnergy, Prm);
138
139
140%  Step 3. Temporal pitch tracking based on NCCF
141%  Calculate a pitch track based on time-domain processing
142%  Pitch tracking from the filterd original signal 
143[TPitch1, TMerit1] = tm_trk(DataB, nFs, SPitch, pStd, pAvg, Prm);
144
145%  Pitch tracking from the filterd nonlinear signal 
146[TPitch2, TMerit2] = tm_trk(DataD, nFs, SPitch, pStd, pAvg, Prm);
147
148% Refine pitch candidates 
149[RPitch, Merit] = refine(TPitch1, TMerit1, TPitch2, TMerit2, SPitch, ...
150                        Energy, VUVEnergy, Prm);
151
152% Step 5. Use dyanamic programming to determine the final pitch
153Pitch  = dynamic(RPitch, Merit, Energy, Prm);
154numfrms = length(Pitch);
155frmrate = Prm.frame_space;
