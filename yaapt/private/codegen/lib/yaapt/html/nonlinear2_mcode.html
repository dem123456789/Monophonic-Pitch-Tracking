1function[DataB, DataC, DataD, newFs] = nonlinear(DataA, Fs, Prm)
 2%NONLINEAR Create the nonlinear processed signal
 3%
 4%   [DataB, DataC, DataD, newFs] = nonlinear(DataA, Fs, Prm) creates
 5%   nonlinear signal (squared signal) and applys filtering.
 6%
 7% INPUTS: 
 8%   DataA: The original signal from read_audio.m 
 9%   Fs:    The sampling rate for the signal
10%   Prm:   Array of parameters used to control performance
11%
12% OUTPUTS:
13%   DataB: The original signal, bandpass filtered with F1.
14%   DataC: The nonlinear signal, no filtering
15%   DataD: The nonlinear signal, bandpass filtered with F1.
16%   newFs: The sampling rate of the new signal 
17
18%   Creation date:  Jun. 30, 2006
19%   Programers:     Hongbing Hu, Princy, Zahorian
20
21%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
22%     This file is a part of the YAAPT program, designed for a fundamental 
23%   frequency tracking algorithm that is extermely robust for both high quality 
24%   and telephone speech.  
25%     The YAAPT program was created by the Speech Communication Laboratory of
26%   the state university of New York at Binghamton. The program is available 
27%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
28%   information about the program could be found at "A spectral/temporal 
29%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
30%   June 2008.
31%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
32
33Fs_min = 1000;    % Do not decimate if Fs less than this
34
35%  Parameters for filtering original signal, with a broader band
36Filter_order = Prm.bp_forder;
37%F_hp = Prm.bp_low;
38%F_lp = Prm.bp_high;
39
40if (Fs > Fs_min)
41    dec_factor = Prm.dec_factor;
42else
43    dec_factor = 1;
44end
45
46
47% Creates the bandpass filters
48lenDataA = length(DataA);
49
50% filter F1
51% w1  = (F_hp / (Fs/2));
52% w2  = (F_lp / (Fs/2));
53  w1  = 0.005;
54  w2  = 0.15;
55  w   = [w1 w2];
56  b_F1 = coder.const(@fir1,Filter_order,w);
57% b_F1 = fir1(Filter_order,w);
58  a   = 1;
59
60
61%filtering the original data with the bandpass filter,
62% DataA   is original acoustic signal
63
64% Original signal filtered with F1
65tempData = filter(b_F1,a,DataA);
66% LenData_dec = fix ((lenDataA+dec_factor-1)/dec_factor);
67DataB = tempData(1:dec_factor:lenDataA);
68
69
70%   Create nonlinear version of signal
71
72%   Original signal
73%   DataC =  DataA;
74%   Absoulte value of the signal
75%   DataC =  abs(DataA);
76%   Squared value of the signal
77    DataC =  DataA .^2;
78
79
80%   Nonlinear version filtered with F1
81tempData = filter(b_F1,a,DataC);
82LenData_dec = fix ((lenDataA+dec_factor-1)/dec_factor);
83DataD = zeros(1, LenData_dec);
84DataD = tempData(1:dec_factor:lenDataA);
85
86newFs = Fs/dec_factor;
87
