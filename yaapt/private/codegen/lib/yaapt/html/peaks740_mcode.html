1function[Pitch, Merit] = peaks(Data, delta, maxpeaks, Prm)
  2%PEAKS  find peaks in SHC 
  3%
  4%   [Pitch,Merit] = peaks(Data, delta, maxpeaks, Prm)
  5%   computes peaks in a frequency domain function
  6%   associated with the peaks found in each frame based on the
  7%   correlation sequence.
  8%
  9%INPUTS:
 10%   Data:  The input signal(either autocorrelated/normalized cross correlated).
 11%   delta: The resolution of spectrum
 12%   maxpeaks: The max number of peaks picked 
 13%   Prm:   Parameters
 14%
 15%OUTPUTS: 
 16%   Pitch:  The Pitch/Pitch values for the peaks found for each frame.
 17%   Merit:  The Merit values of the peaks found in each frame.
 18
 19%  Creation date:   March 1, 2006
 20%  Revision dates:  March 11, 2006,  Jun 26, 2006, July 27, 2007
 21%  Programmer: S.A.Zahorian, Princy  Dikshit, Hongbing Hu
 22
 23%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 24%     This file is a part of the YAAPT program, designed for a fundamental 
 25%   frequency tracking algorithm that is extermely robust for both high quality 
 26%   and telephone speech.  
 27%     The YAAPT program was created by the Speech Communication Laboratory of
 28%   the state university of New York at Binghamton. The program is available 
 29%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 30%   information about the program could be found at "A spectral/temporal 
 31%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 32%   June 2008.
 33%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 34
 35%-- PARAMETERS ----------------------------------------------------------------
 36% Threshold for max avaliable peak
 37PEAK_THRESH1 = Prm.shc_thresh1;
 38    
 39% Threshold for available peaks
 40PEAK_THRESH2 = Prm.shc_thresh2;
 41
 42%-- COMPUTED VARIABLES --------------------------------------------------------
 43epsilon = .00000000000001;
 44
 45% Length in Hz of range(must be largest to be a peak)
 46width   = fix(Prm.shc_pwidth/delta);     % Window width in sample
 47if (mod(width,2)==0)
 48    width = width + 1;
 49end
 50% The center of the window is defined as the peak location.
 51center = ceil(width/2); 
 52
 53% Lowest frequency at which F0 is allowed
 54min_lag  = fix(Prm.f0_min/delta - width/4);
 55% Highest frequency at which F0 is allowed
 56max_lag  = fix(Prm.f0_max/delta + width/4);
 57
 58if (min_lag < 1)
 59    min_lag = 1;
 60    warning('min_lag is too low and adjusted (%d)', min_lag); 
 61end
 62if max_lag > (length(Data) - width)
 63    max_lag = length(Data) - width;
 64    warning('max_lag is too high and adjusted (%d)', max_lag);
 65end
 66
 67
 68%-- INITIALIZATION -----------------------------------------------------------
 69Pitch     = zeros(1, maxpeaks);         % Peak(Pitch) candidates
 70Merit     = zeros(1, maxpeaks);         % Merits for peaks
 71
 72%-- MAIN ROUTINE --------------------------------------------------------------
 73% Normalize the signal so that peak value = 1
 74max_data = max(Data(min_lag:max_lag));
 75if (max_data > epsilon)
 76    Data = Data/max_data;
 77end
 78
 79% If true there are no large peaks and we assume that signal is unvoiced
 80avg_data = mean(Data(min_lag:max_lag));
 81if (avg_data > 1/PEAK_THRESH1)    
 82%    numpeaks = 0;
 83    Pitch    = zeros(1, maxpeaks);
 84    Merit    = ones (1, maxpeaks);
 85    % force an early end for unoviced frame 
 86    return
 87end
 88
 89% Step 1
 90% Find all peaks for search range
 91% a "peak" must be the higher than a specfied number of
 92% points on either side of point.  Peaks are further "cleaned"
 93% up, to retain only best peaks i.e. peaks which do not meet certain
 94% criteria are eliminated.
 95
 96numpeaks = 0;
 97for n = min_lag:max_lag
 98    [y, lag]  = max(Data(n:(n+width-1)));
 99    % find peaks which are larger than threshold   
100    if (lag == center) && y>(PEAK_THRESH2*avg_data)
101
102        % Note Pitch(1) = delta, Pitch(2) = 2*delta
103        % Convert FFT indices to Pitch in Hz
104        numpeaks = numpeaks + 1;
105        Pitch (numpeaks)  = (n+center-1)*delta;
106        Merit(numpeaks)  = y;
107    end
108end
109
110% Step 2
111% Be sure there is large peak
112if (max(Merit)/avg_data < PEAK_THRESH1)
113%   numpeaks = 0;
114    Pitch      = zeros(1, maxpeaks);
115    Merit     = ones (1, maxpeaks);
116    return
117end
118    
119% Step 3    
120% Order the peaks according to size,  considering at most maxpeaks
121[Merit, Idx] = sort(Merit, 'descend');
122Pitch = Pitch(Idx);
123% keep the number of peaks not greater than max number
124numpeaks = min(numpeaks, maxpeaks);
125Pitch  = Pitch(1:numpeaks);
126Merit = Merit(1:numpeaks);
127
128% Step 4
129% Insert candidates to reduce pitch doubling and pitch halving, if needed
130if (numpeaks > 0)
131    % if best peak has F < this, insert peak at 2F
132    if (Pitch(1) > Prm.f0_double)   
133        numpeaks = min(numpeaks+1, maxpeaks);
134        Pitch(numpeaks) = Pitch(1)/2.0;
135        % Set merit for inserted peaks
136        Merit(numpeaks) = Prm.merit_extra;
137    end
138
139    % If best peak has F > this, insert peak at half F
140    if (Pitch(1) < Prm.f0_half)     
141        numpeaks = min(numpeaks+1, maxpeaks);
142        Pitch(numpeaks) = 2.0*Pitch(1);
143        Merit(numpeaks) = Prm.merit_extra;
144    end
145    
146    % Fill in  frames with less than maxpeaks with best choice
147    if (numpeaks < maxpeaks)
148        Pitch(numpeaks+1:maxpeaks)  = Pitch(1);
149        Merit(numpeaks+1:maxpeaks) = Merit(1);
150    end
151else 
152    Pitch    = zeros(1, maxpeaks);
153    Merit    = ones (1, maxpeaks);
154end      
155    
156%==============================================================================
