1function[Pitch, Merit] = peaks(Data, delta, maxpeaks, Prm)
  2%PEAKS  find peaks in SHC 
  3%
  4%   [Pitch,Merit] = peaks(Data, delta, maxpeaks, Prm)
  5%   computes peaks in a frequency domain function
  6%   associated with the peaks found in each frame based on the
  7%   correlation sequence.
  8%
  9%INPUTS:
 10%   Data:  The input signal(either autocorrelated/normalized cross correlated).
 11%   delta: The resolution of spectrum
 12%   maxpeaks: The max number of peaks picked 
 13%   Prm:   Parameters
 14%
 15%OUTPUTS: 
 16%   Pitch:  The Pitch/Pitch values for the peaks found for each frame.
 17%   Merit:  The Merit values of the peaks found in each frame.
 18
 19%  Creation date:   March 1, 2006
 20%  Revision dates:  March 11, 2006,  Jun 26, 2006, July 27, 2007
 21%  Programmer: S.A.Zahorian, Princy  Dikshit, Hongbing Hu
 22
 23%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 24%     This file is a part of the YAAPT program, designed for a fundamental 
 25%   frequency tracking algorithm that is extermely robust for both high quality 
 26%   and telephone speech.  
 27%     The YAAPT program was created by the Speech Communication Laboratory of
 28%   the state university of New York at Binghamton. The program is available 
 29%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 30%   information about the program could be found at "A spectral/temporal 
 31%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 32%   June 2008.
 33%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 34
 35%-- PARAMETERS ----------------------------------------------------------------
 36% Threshold for max avaliable peak
 37PEAK_THRESH1 = Prm.shc_thresh1;
 38    
 39% Threshold for available peaks
 40PEAK_THRESH2 = Prm.shc_thresh2;
 41
 42%-- COMPUTED VARIABLES --------------------------------------------------------
 43epsilon = .00000000000001;
 44
 45% Length in Hz of range(must be largest to be a peak)
 46width   = fix(Prm.shc_pwidth/delta);     % Window width in sample
 47if (mod(width,2)==0)
 48    width = width + 1;
 49end
 50% The center of the window is defined as the peak location.
 51center = ceil(width/2); 
 52
 53% Lowest frequency at which F0 is allowed
 54min_lag  = fix(Prm.f0_min/delta - width/4);
 55% Highest frequency at which F0 is allowed
 56max_lag  = fix(Prm.f0_max/delta + width/4);
 57
 58% if (min_lag < 1)
 59%     min_lag = 1;
 60%     disp('min_lag is too low and adjusted (%d)', min_lag); 
 61% end
 62% if max_lag > (length(Data) - width)
 63%     max_lag = length(Data) - width;
 64%     disp('max_lag is too high and adjusted (%d)', max_lag);
 65% end
 66
 67
 68%-- INITIALIZATION -----------------------------------------------------------
 69max_allow_lags = 100;
 70Pitch     = zeros(1, max_allow_lags);         % Peak(Pitch) candidates
 71Merit     = zeros(1, max_allow_lags);         % Merits for peaks
 72
 73%-- MAIN ROUTINE --------------------------------------------------------------
 74% Normalize the signal so that peak value = 1
 75max_data = max(Data(min_lag:max_lag));
 76if (max_data > epsilon)
 77    Data = Data/max_data;
 78end
 79
 80% If true there are no large peaks and we assume that signal is unvoiced
 81avg_data = mean(Data(min_lag:max_lag));
 82if (avg_data > 1/PEAK_THRESH1)    
 83%    numpeaks = 0;
 84    Pitch    = zeros(1, maxpeaks);
 85    Merit    = ones (1, maxpeaks);
 86    % force an early end for unoviced frame 
 87    return
 88end
 89
 90% Step 1
 91% Find all peaks for search range
 92% a "peak" must be the higher than a specfied number of
 93% points on either side of point.  Peaks are further "cleaned"
 94% up, to retain only best peaks i.e. peaks which do not meet certain
 95% criteria are eliminated.
 96
 97numpeaks = 0;
 98for n = min_lag:max_lag
 99    [y, lag]  = max(Data(n:(n+width-1)));
100    % find peaks which are larger than threshold   
101    if (lag == center) && y>(PEAK_THRESH2*avg_data)
102
103        % Note Pitch(1) = delta, Pitch(2) = 2*delta
104        % Convert FFT indices to Pitch in Hz
105        numpeaks = numpeaks + 1;
106        Pitch(numpeaks)  = (n+center-1)*delta;
107        Merit(numpeaks)  = y;
108    end
109end
110
111% Step 2
112% Be sure there is large peak
113if (max(Merit)/avg_data < PEAK_THRESH1)
114%   numpeaks = 0;
115    Pitch      = zeros(1, maxpeaks);
116    Merit     = ones (1, maxpeaks);
117    return
118end
119    
120% Step 3    
121% Order the peaks according to size,  considering at most maxpeaks
122[Merit, Idx] = sort(Merit, 'descend');
123Pitch = Pitch(Idx);
124% keep the number of peaks not greater than max number
125numpeaks = min(numpeaks, maxpeaks);
126
127Pitch  = Pitch(1:numpeaks);
128Merit = Merit(1:numpeaks);
129% Step 4
130% Insert candidates to reduce pitch doubling and pitch halving, if needed
131if (numpeaks > 0)
132    % if best peak has F < this, insert peak at 2F
133    if (Pitch(1) > Prm.f0_double)   
134        numpeaks = min(numpeaks+1, maxpeaks);
135        if(numpeaks>length(Pitch))
136            Pitch = [Pitch zeros(1,numpeaks-length(Pitch))];
137            Merit = [Merit zeros(1,numpeaks-length(Merit))];
138        end
139        Pitch(numpeaks) = Pitch(1)/2.0;
140        % Set merit for inserted peaks
141        Merit(numpeaks) = Prm.merit_extra;
142    end
143
144    % If best peak has F > this, insert peak at half F
145    if (Pitch(1) < Prm.f0_half)     
146        numpeaks = min(numpeaks+1, maxpeaks);
147        if(numpeaks>length(Pitch))
148            Pitch = [Pitch zeros(1,numpeaks-length(Pitch))];
149            Merit = [Merit zeros(1,numpeaks-length(Merit))];
150        end
151        Pitch(numpeaks) = 2.0*Pitch(1);
152        Merit(numpeaks) = Prm.merit_extra;
153    end
154    
155    % Fill in  frames with less than maxpeaks with best choice
156    if (numpeaks < maxpeaks)
157        Pitch = [Pitch zeros(1,maxpeaks-numpeaks)];
158        Merit = [Merit zeros(1,maxpeaks-numpeaks)];
159        Pitch(numpeaks+1:maxpeaks)  = Pitch(1);
160        Merit(numpeaks+1:maxpeaks) = Merit(1);
161    end
162else 
163    Pitch    = zeros(1, maxpeaks);
164    Merit    = ones (1, maxpeaks);
165end      
166    
167%==============================================================================
