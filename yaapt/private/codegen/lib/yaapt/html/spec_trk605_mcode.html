1function[SPitch, VUVSPitch, pAvg, pStd]= spec_trk(Data, Fs, VUVEnergy, Prm)
  2%SPEC_TRK  Spectral pitch tracking for YAAPT pitch tracking
  3%
  4%   [SPitch, VUVSPitch, pAvg, pStd]=spec_trk(Data, Fs, VUVEnergy, Prm) 
  5%   computes estimates of pitch  using nonlinearly processed
  6%   speech (typically square or absolute value) and frequency domain processing
  7%   Search for frequencies which have energy at multiplies of that frequency
  8%
  9%INPUTS:
 10%   Data:      Nonlinearly processed signal, filtered with F1. (~50 to 1500Hz)
 11%   Fs:        The sampling frequency.
 12%   VUVEnergy: Voiced/ unvoiced decision, depending on nlfer
 13%   Prm:       Parameters
 14%
 15%OUTPUTS:
 16%   SPitch:    The spectral Pitch track, with the unvoiced regions filled using interpolation.
 17%   VUVSPitch: The spectral Pitch track, with  unvoiced regions set at zero
 18%   pAvg:      The average of the Pitch track value.
 19%   pStd:      The standard deviation in the track.
 20
 21%   Creation date:  Feb 20, 2006
 22%   Revision dates: Feb 22, 2006,  March 11, 2006, April 5, 2006,
 23%                   Jun 25, 2006,  June 13, 2007
 24%   Programers:     Hongbing Hu, Princy, Zahorian
 25
 26%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 27%     This file is a part of the YAAPT program, designed for a fundamental 
 28%   frequency tracking algorithm that is extermely robust for both high quality 
 29%   and telephone speech.  
 30%     The YAAPT program was created by the Speech Communication Laboratory of
 31%   the state university of New York at Binghamton. The program is available 
 32%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 33%   information about the program could be found at "A spectral/temporal 
 34%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 35%   June 2008.
 36%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 37
 38%-- PARAMETERS: set up all of these parameters --------------------------------
 39nframesize = fix(Prm.frame_length*Fs/1000);    
 40nframejump = fix(Prm.frame_space*Fs/1000); 
 41noverlap= nframesize-nframejump;         % overlap width in sample 
 42numframes = fix((length(Data)-noverlap)/nframejump);
 43
 44% use larger frame size
 45nframesize = nframesize*2;
 46
 47% Max number of peak candidates found
 48maxpeaks = Prm.shc_maxpeaks;
 49nfftlength = Prm.fft_length;                       % FFT length
 50% Resolution of spectrum
 51delta         = Fs/nfftlength;               
 52% Window width in sample
 53window_length = fix(Prm.shc_window/delta);
 54if (mod(window_length,2)==0)
 55    window_length = window_length + 1;
 56end
 57% Half window width
 58half_winlen   = fix(Prm.shc_window/(delta*2)); 
 59% Max range of SHC 
 60max_SHC   = fix((Prm.f0_max+Prm.shc_pwidth*2)/delta);
 61% Min range of SHC 
 62min_SHC   = ceil(Prm.f0_min/delta);       
 63% Number of harmomics considered
 64numharmonics = Prm.shc_numharms;
 65
 66
 67%-- INITIALIZATION -----------------------------------------------------------
 68CandsPitch = zeros(maxpeaks, numframes);
 69CandsMerit = ones(maxpeaks, numframes);
 70% Zero padding
 71Data(end:(numframes-1)*nframejump+nframesize) = 0;
 72
 73%-- MAIN ROUTINE --------------------------------------------------------------
 74% Compute SHC for voiced frame
 75Kaiser_window = Mykaiser(nframesize);
 76% Kaiser_window_test = kaiser(nframesize);
 77SHC = zeros(1,max_SHC);
 78
 79winix = repmat([1:window_length], numharmonics+1,1);
 80rowix = repmat([1:numharmonics+1]', 1, window_length);
 81
 82for frame = 1:numframes
 83    if (VUVEnergy(frame) > 0)
 84        firstp = 1+(frame-1)*(nframejump);
 85        Signal = Data(firstp:firstp+nframesize-1) .* Kaiser_window;
 86        Signal = Signal - mean(Signal);
 87%          Magnit = [zeros(half_winlen, 1); abs(fft(Signal , nfftlength))];
 88         Magnit = [zeros(half_winlen, 1); abs(fft(Signal , nfftlength))'];
 89
 90        % Compute SHC (Spectral Harmonic Correlation)
 91        for k=min_SHC:max_SHC;
 92            SHC(k) = sum(prod(Magnit( winix + k*rowix )));
 93        end
 94        [CandsPitch(:,frame), CandsMerit(:,frame)]=peaks(SHC,delta, maxpeaks, Prm);
 95    else 
 96        % if energy is low,  let frame be considered as unvoiced
 97        CandsPitch(:,frame) = zeros(1,maxpeaks);
 98        CandsMerit(:,frame) = ones(1, maxpeaks);
 99    end
100end
101
102% display('Running new code');
103
104% for frame = 1:numframes
105%     if (VUVEnergy(frame) > 0)
106%         firstp = 1+(frame-1)*(nframejump);
107%         Signal = Data(firstp:firstp+nframesize-1) .* Kaiser_window;
108%         Signal = Signal - mean(Signal);
109%         Magnit = abs(fft(Signal , nfftlength));
110%         
111%         % Compute SHC (Spectral Harmonic Correlation)
112%         for k=min_SHC:max_SHC;
113%             MultHarms = ones(1, window_length);
114% 
115%             % Set each harmonics, 1=f0 2=2*f0 etc
116%             for n=1:numharmonics+1  % Number of harmomics considered
117%                 nstart = n*k-half_winlen;
118%                 if nstart < 0
119%                     Harm =  zeros(1, window_length);
120%                     Harm(abs(nstart)+1:window_length)=Magnit(1:nstart+window_length);
121%                 else
122%                     Harm(1:window_length)=Magnit(nstart+1:nstart+window_length);
123%                 end
124%                 MultHarms = MultHarms .* Harm;
125%             end
126%             SHC(k) = sum(MultHarms);
127%         end
128%         [CandsPitch(:,frame), CandsMerit(:,frame)]=peaks(SHC,delta, maxpeaks, Prm);
129%     else 
130%         % if energy is low,  let frame be considered as unvoiced
131%         CandsPitch(:,frame) = zeros(1,maxpeaks);
132%         CandsMerit(:,frame) = ones(1, maxpeaks);
133%     end
134% end
135
136% Extract the Pitch candidates of voiced frames for the future Pitch selection 
137SPitch = CandsPitch(1,:);
138Idx_voiced = SPitch > 0; 
139VCandsPitch  = CandsPitch(:,Idx_voiced);
140VCandsMerit = CandsMerit(:,Idx_voiced);
141Num_VCands = sum(Idx_voiced);
142
143% Average, STD of the first choice candidates
144avg_voiced = mean(VCandsPitch(1,:));
145std_voiced = std(VCandsPitch(1,:));
146
147% Weight the deltas, so that higher merit candidates are considered
148% more favorably
149delta1 = abs((VCandsPitch - 0.8*avg_voiced)).*(3-VCandsMerit);
150
151% Interpolation of the weigthed candidates
152[VR, Idx] = min(delta1); 
153VPeak_minmrt  = zeros(1, length(Idx));
154VMerit_minmrt = zeros(1, length(Idx));
155for n=1: length(Idx)
156    VPeak_minmrt(n)  = VCandsPitch(Idx(n), n);
157    VMerit_minmrt(n) = VCandsMerit(Idx(n), n); 
158end
159VPeak_minmrt = Mymedfilt1(VPeak_minmrt, max(1,Prm.median_value-2));
160% VPeak_minmrt_test = medfilt1(VPeak_minmrt, max(1,Prm.median_value-2));
161% Replace the lowest merit candidates by the median smoothed ones
162% computed from highest merit peaks above
163for n=1: length(Idx)
164    VCandsPitch(Idx(n), n) = VPeak_minmrt(n);
165    % Assign merit for the smoothed peaks
166    VCandsMerit(Idx(n), n) = VMerit_minmrt(n);
167end
168
169% Use dynamic programming to find best overal path among pitch candidates
170% Dynamic weight for transition costs
171% balance between local and transition costs
172weight_trans = Prm.dp5_k1*std_voiced/avg_voiced;
173
174if (Num_VCands >2) 
175  [VPitch] = dynamic5(VCandsPitch, VCandsMerit,weight_trans);
176   VPitch = Mymedfilt1(VPitch, max(Prm.median_value-2, 1));
177%    VPitch_test = medfilt1(VPitch, max(Prm.median_value-2, 1));
178   
179else
180     
181    if (Num_VCands > 0) 
182      for i = 1:Num_VCands   % assume at least 1 voiced candidate
183       VPitch (i) = 150;   %   Not really correct,  but should prevent hang up if only  0,1,2,3 voiced candidates
184      end; 
185                %   This should only occur for very short pitch tracks, and even then rarely
186    else
187        VPitch(1) = 150;    % to handle when no real voiced candidates
188        Idx_voiced(1) = 1;
189    end;
190    
191end;       
192
193
194% Computing some statistics from the voiced frames
195pAvg = mean(VPitch);
196pStd = std(VPitch);
197
198% Streching out the smoothed pitch track
199SPitch(Idx_voiced) = VPitch;
200
201% Interpolating thru unvoiced frames
202if (SPitch(1) < pAvg/2) SPitch(1)   = pAvg;   end;
203if (SPitch(end) <pAvg/2) SPitch(end) = pAvg;  end;
204
205[Indrows, Indcols, Vals] = find(SPitch);
206SPitch = interp1(Indcols, Vals, [1:numframes], 'pchip');
207
208FILTER_ORDER = 3;
209flt_b = (ones(1,FILTER_ORDER))/FILTER_ORDER ;
210flt_a = 1;
211
212SPitch = filter(flt_b, flt_a, SPitch);
213
214
215%  above messes up  first few values of SPitch  ---  simple fix up
216%  Note--   this fix up should be based on above filter order
217
218SPitch(1) = SPitch(3);
219SPitch(2) = SPitch (4);
220
221% Create pitch track with Voiced/Unvoiced decision
222VUVSPitch = SPitch;
223VUVSPitch(VUVEnergy==0) = 0;
224
