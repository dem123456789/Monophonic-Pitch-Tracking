1function[SPitch, VUVSPitch, pAvg, pStd]= spec_trk(Data, Fs, VUVEnergy, Prm)
  2%SPEC_TRK  Spectral pitch tracking for YAAPT pitch tracking
  3%
  4%   [SPitch, VUVSPitch, pAvg, pStd]=spec_trk(Data, Fs, VUVEnergy, Prm) 
  5%   computes estimates of pitch  using nonlinearly processed
  6%   speech (typically square or absolute value) and frequency domain processing
  7%   Search for frequencies which have energy at multiplies of that frequency
  8%
  9%INPUTS:
 10%   Data:      Nonlinearly processed signal, filtered with F1. (~50 to 1500Hz)
 11%   Fs:        The sampling frequency.
 12%   VUVEnergy: Voiced/ unvoiced decision, depending on nlfer
 13%   Prm:       Parameters
 14%
 15%OUTPUTS:
 16%   SPitch:    The spectral Pitch track, with the unvoiced regions filled using interpolation.
 17%   VUVSPitch: The spectral Pitch track, with  unvoiced regions set at zero
 18%   pAvg:      The average of the Pitch track value.
 19%   pStd:      The standard deviation in the track.
 20
 21%   Creation date:  Feb 20, 2006
 22%   Revision dates: Feb 22, 2006,  March 11, 2006, April 5, 2006,
 23%                   Jun 25, 2006,  June 13, 2007
 24%   Programers:     Hongbing Hu, Princy, Zahorian
 25
 26%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 27%     This file is a part of the YAAPT program, designed for a fundamental 
 28%   frequency tracking algorithm that is extermely robust for both high quality 
 29%   and telephone speech.  
 30%     The YAAPT program was created by the Speech Communication Laboratory of
 31%   the state university of New York at Binghamton. The program is available 
 32%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
 33%   information about the program could be found at "A spectral/temporal 
 34%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
 35%   June 2008.
 36%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 37
 38%-- PARAMETERS: set up all of these parameters --------------------------------
 39nframesize = fix(Prm.frame_length*Fs/1000);    
 40nframejump = fix(Prm.frame_space*Fs/1000); 
 41noverlap= nframesize-nframejump;         % overlap width in sample 
 42numframes = fix((length(Data)-noverlap)/nframejump);
 43
 44% use larger frame size
 45nframesize = nframesize*2;
 46
 47% Max number of peak candidates found
 48maxpeaks = Prm.shc_maxpeaks;
 49nfftlength = Prm.fft_length;                       % FFT length
 50% Resolution of spectrum
 51delta         = Fs/nfftlength;               
 52% Window width in sample
 53window_length = fix(Prm.shc_window/delta);
 54if (mod(window_length,2)==0)
 55    window_length = window_length + 1;
 56end
 57% Half window width
 58half_winlen   = fix(Prm.shc_window/(delta*2)); 
 59% Max range of SHC 
 60max_SHC   = fix((Prm.f0_max+Prm.shc_pwidth*2)/delta);
 61% Min range of SHC 
 62min_SHC   = ceil(Prm.f0_min/delta);       
 63% Number of harmomics considered
 64numharmonics = Prm.shc_numharms;
 65
 66
 67%-- INITIALIZATION -----------------------------------------------------------
 68CandsPitch = zeros(maxpeaks, numframes);
 69CandsMerit = ones(maxpeaks, numframes);
 70% Zero paddinga
 71% a = Data;
 72% Data(end:(numframes-1)*nframejump+nframesize) = 0;
 73Data(end) = 0;
 74zero_padded = (numframes-1)*nframejump+nframesize-length(Data);
 75if(zero_padded >0)    
 76    Data = [Data zero_padded];
 77end
 78%-- MAIN ROUTINE --------------------------------------------------------------
 79% Compute SHC for voiced frame
 80Kaiser_window = Mykaiser(nframesize);
 81% Kaiser_window_test = kaiser(nframesize);
 82SHC = zeros(1,max_SHC);
 83
 84winix = repmat([1:window_length], numharmonics+1,1);
 85rowix = repmat([1:numharmonics+1]', 1, window_length);
 86
 87for frame = 1:numframes
 88    if (VUVEnergy(frame) > 0)
 89        firstp = 1+(frame-1)*(nframejump);
 90        Signal = Data(firstp:firstp+nframesize-1) .* Kaiser_window';
 91        Signal = Signal - mean(Signal);
 92%            Magnit = [zeros(half_winlen, 1); abs(fft(Signal , nfftlength))];
 93             Magnit = [zeros(half_winlen, 1); abs(fft(Signal , nfftlength))'];
 94
 95        % Compute SHC (Spectral Harmonic Correlation)
 96        for k=min_SHC:max_SHC;
 97            SHC(k) = sum(prod(Magnit( winix + k*rowix )));
 98        end
 99        [CandsPitch(:,frame), CandsMerit(:,frame)]=peaks(SHC,delta, maxpeaks, Prm);
100    else 
101        % if energy is low,  let frame be considered as unvoiced
102        CandsPitch(:,frame) = zeros(1,maxpeaks);
103        CandsMerit(:,frame) = ones(1, maxpeaks);
104    end
105end
106
107% display('Running new code');
108
109% for frame = 1:numframes
110%     if (VUVEnergy(frame) > 0)
111%         firstp = 1+(frame-1)*(nframejump);
112%         Signal = Data(firstp:firstp+nframesize-1) .* Kaiser_window;
113%         Signal = Signal - mean(Signal);
114%         Magnit = abs(fft(Signal , nfftlength));
115%         
116%         % Compute SHC (Spectral Harmonic Correlation)
117%         for k=min_SHC:max_SHC;
118%             MultHarms = ones(1, window_length);
119% 
120%             % Set each harmonics, 1=f0 2=2*f0 etc
121%             for n=1:numharmonics+1  % Number of harmomics considered
122%                 nstart = n*k-half_winlen;
123%                 if nstart < 0
124%                     Harm =  zeros(1, window_length);
125%                     Harm(abs(nstart)+1:window_length)=Magnit(1:nstart+window_length);
126%                 else
127%                     Harm(1:window_length)=Magnit(nstart+1:nstart+window_length);
128%                 end
129%                 MultHarms = MultHarms .* Harm;
130%             end
131%             SHC(k) = sum(MultHarms);
132%         end
133%         [CandsPitch(:,frame), CandsMerit(:,frame)]=peaks(SHC,delta, maxpeaks, Prm);
134%     else 
135%         % if energy is low,  let frame be considered as unvoiced
136%         CandsPitch(:,frame) = zeros(1,maxpeaks);
137%         CandsMerit(:,frame) = ones(1, maxpeaks);
138%     end
139% end
140
141% Extract the Pitch candidates of voiced frames for the future Pitch selection 
142SPitch = CandsPitch(1,:);
143Idx_voiced = SPitch > 0; 
144VCandsPitch  = CandsPitch(:,Idx_voiced);
145VCandsMerit = CandsMerit(:,Idx_voiced);
146Num_VCands = sum(Idx_voiced);
147
148% Average, STD of the first choice candidates
149avg_voiced = mean(VCandsPitch(1,:));
150std_voiced = std(VCandsPitch(1,:));
151
152% Weight the deltas, so that higher merit candidates are considered
153% more favorably
154delta1 = abs((VCandsPitch - 0.8*avg_voiced)).*(3-VCandsMerit);
155
156% Interpolation of the weigthed candidates
157[VR, Idx] = min(delta1); 
158VPeak_minmrt  = zeros(1, length(Idx));
159VMerit_minmrt = zeros(1, length(Idx));
160for n=1: length(Idx)
161    VPeak_minmrt(n)  = VCandsPitch(Idx(n), n);
162    VMerit_minmrt(n) = VCandsMerit(Idx(n), n); 
163end
164VPeak_minmrt = Mymedfilt1(VPeak_minmrt, max(1,Prm.median_value-2));
165% VPeak_minmrt_test = medfilt1(VPeak_minmrt, max(1,Prm.median_value-2));
166% Replace the lowest merit candidates by the median smoothed ones
167% computed from highest merit peaks above
168for n=1: length(Idx)
169    VCandsPitch(Idx(n), n) = VPeak_minmrt(n);
170    % Assign merit for the smoothed peaks
171    VCandsMerit(Idx(n), n) = VMerit_minmrt(n);
172end
173
174% Use dynamic programming to find best overal path among pitch candidates
175% Dynamic weight for transition costs
176% balance between local and transition costs
177weight_trans = Prm.dp5_k1*std_voiced/avg_voiced;
178VPitch = [];
179if (Num_VCands >2) 
180  [VPitch] = dynamic5(VCandsPitch, VCandsMerit,weight_trans);
181   VPitch = Mymedfilt1(VPitch, max(Prm.median_value-2, 1));
182%    VPitch_test = medfilt1(VPitch, max(Prm.median_value-2, 1));
183   
184else
185     
186    if (Num_VCands > 0) 
187%       for i = 1:Num_VCands   % assume at least 1 voiced candidate
188%        VPitch (i) = 150;   %   Not really correct,  but should prevent hang up if only  0,1,2,3 voiced candidates
189%       end; 
190        VPitch(1:Num_VCands) = 150;
191                %   This should only occur for very short pitch tracks, and even then rarely
192    else
193        VPitch(1) = 150;    % to handle when no real voiced candidates
194        Idx_voiced(1) = 1;
195    end;
196    
197end;       
198
199
200% Computing some statistics from the voiced frames
201pAvg = mean(VPitch,2);
202pStd = std(VPitch,0,2);
203
204% Streching out the smoothed pitch track
205SPitch(Idx_voiced) = VPitch;
206
207% Interpolating thru unvoiced frames
208SPitch_temp_first = SPitch(1);
209SPitch_temp_end = SPitch(end);
210if all(SPitch_temp_first < (pAvg/2))
211    SPitch(1)   = pAvg;   
212end
213if all(SPitch_temp_end < (pAvg/2))
214    SPitch(end) = pAvg;  
215end
216
217[Indrows, Indcols, Vals] = find(SPitch);
218SPitch = interp1(Indcols, Vals, [1:numframes], 'pchip');
219
220FILTER_ORDER = 3;
221flt_b = (ones(1,FILTER_ORDER))/FILTER_ORDER ;
222flt_a = 1;
223
224SPitch = filter(flt_b, flt_a, SPitch);
225
226
227%  above messes up  first few values of SPitch  ---  simple fix up
228%  Note--   this fix up should be based on above filter order
229
230SPitch(1) = SPitch(3);
231SPitch(2) = SPitch (4);
232
233% Create pitch track with Voiced/Unvoiced decision
234VUVSPitch = SPitch;
235VUVSPitch(VUVEnergy==0) = 0;
236
237% SPitch = SPitch - delta;
