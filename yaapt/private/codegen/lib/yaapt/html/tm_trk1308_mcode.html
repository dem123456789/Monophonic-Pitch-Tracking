1function [Pitch, Merit] = tm_trk(Data, Fs, SPitch, pStd, pAvg, Prm)
 2%TM_TRK  Temporal pitch tracking for YAAPT pitch tracking 
 3%
 4%   [Pitch, Merit] = tm_trk(Data, Fs, SPitch, pStd, pAvg, Prm)
 5%   computes estimates of pitch  using nonlinearly processed
 6%   speech (typically square or absolute value) and frequency domain processing
 7%   Search for frequencies which have energy at multiplies of that frequency
 8%
 9%INPUTS:
10%   Data:   Nonlinearly processed signal
11%   Fs:    The sampling frequency.
12%   SPitch:The spectral F0 track, with the unvoiced
13%   pStd:  The STD of voiced spectral F0 track .
14%   pAvg:  The mean of voiced spectral F0 track .
15%   Prm:   Array of parameters used to control performance
16%
17%OUTPUTS:
18%   Pitch: Computed pitch candidates array
19%   Merit: The merit of array of pitch candidates
20
21%   Creation :   Jun 20, 2006, July 27, 2007
22%   Programers:  Hongbing Hu
23
24%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
25%     This file is a part of the YAAPT program, designed for a fundamental 
26%   frequency tracking algorithm that is extermely robust for both high quality 
27%   and telephone speech.  
28%     The YAAPT program was created by the Speech Communication Laboratory of
29%   the state university of New York at Binghamton. The program is available 
30%   at http://www.ws.binghamton.edu/zahorian as free software. Further 
31%   information about the program could be found at "A spectral/temporal 
32%   method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6), 
33%   June 2008.
34%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
35
36%-- PARAMETERS ----------------------------------------------------------------
37nframesize = fix(Prm.frame_length*Fs/1000);    
38nframejump = fix(Prm.frame_space*Fs/1000); 
39noverlap   = nframesize-nframejump;         % overlap width in sample 
40numframes  = fix((length(Data)-noverlap)/nframejump);
41
42merit_boost = Prm.merit_boost;
43% The maximum number of pitch candidates
44maxcands   = Prm.nccf_maxcands; 
45%freq_thresh = pAvg;  % seems too large
46freq_thresh = 5 * pStd;
47SRange = zeros(2,length(SPitch));
48% Determine the search range according spectral pitch track
49SRange(1,:)  = max((SPitch-2*pStd), Prm.f0_min);
50SRange(2,:)  = min((SPitch+2*pStd), Prm.f0_max);    
51
52%-- INITIALIZATION -----------------------------------------------------------
53Pitch = zeros(maxcands, numframes);
54Merit = zeros(maxcands, numframes);
55
56%-- MAIN ROUTINE --------------------------------------------------------------
57for n = 1:numframes
58
59    firstp = 1+((n-1)*(nframejump));
60    Signal = Data(firstp:firstp+nframesize-1);
61    
62    % Compute pitch candidates,  and correspnding "merit" values,
63    % up to maxcands per frame Merit values are normalized (0 to
64    % 1),  and depend mainly on size of correlation peaks
65    Lag_min0 = fix(Fs/SRange(2,n)) - 3;
66    Lag_max0 = fix(Fs/SRange(1,n)) + 3;
67    % Compute correaltion
68    Phi =  crs_corr(Signal, Lag_min0, Lag_max0);
69    
70    % The Max_cand pitch candidates are collected into Pitch and
71    % Merit arraies 
72    [Pitch(:,n),Merit(:,n)] = cmp_rate(Phi,Fs,maxcands,Lag_min0,Lag_max0,Prm);
73end 
74
75%   The following lines increase merit for peaks which are very
76%   close to in frequency to those peaks which are close to
77%   smoothed F0 track from spectrogram,   and decrease merit for peaks
78%   which are not close to the smoothed F0 track obtained from
79%   spectrogram
80for i = 1:maxcands
81    diff   =  abs( (Pitch(i,:) - SPitch(1,:)));
82    match1 =  (diff < freq_thresh);
83    match  =  ((1 - diff/freq_thresh) .* match1);
84    Merit(i,:) = ((1+merit_boost)*Merit(i,:)).*match;
85end
86
