  490     /*                       index correponds to  highest merit */
  491     /*    Merit:  Merit array with elements arranged as for pitch */
  492     /*    Energy: NLFER energy ratio from the low frequency regions of spectrogram. */
  493     /*    Prm:    Array of parameters used to control performance */
  494     /*  */
  495     /* OUTPUTS:   */
  496     /*    FinPitch: The final pitch track for the entire utterance pitch */
  497     /*    values are in HZ.Unvoiced frames are assigned 0. */
  498     /*  Note:   This routine is intended for more through testing of */
  499     /*     some thresholds used in dynamic6.   All the dp constants are */
  500     /*     set to fixed values that worked well in testing */
  501     /*     then the dp constants are used to specify the thresholds and other */
  502     /*      constants */
  503     /*    Creation date:   Spring 2001 */
  504     /*    Revision date:   January 3, 2002   June 16, 2007, July 27, 2007 */
  505     /*    Programmers: S. Zahorian,Kavita Kasi, Lingyun Gu, Hongbing Hu */
  506     /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
  507     /*      This file is a part of the YAAPT program, designed for a fundamental  */
  508     /*    frequency tracking algorithm that is extermely robust for both high quality  */
  509     /*    and telephone speech.   */
  510     /*      The YAAPT program was created by the Speech Communication Laboratory of */
  511     /*    the state university of New York at Binghamton. The program is available  */
  512     /*    at http://www.ws.binghamton.edu/zahorian as free software. Further  */
  513     /*    information about the program could be found at "A spectral/temporal  */
  514     /*    method for robust fundamental frequency tracking," J.Acosut.Soc.Am. 123(6),  */
  515     /*    June 2008. */
  516     /* %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% */
  517     /* -- INITIALIZATION ----------------------------------------------------------- */
  518     numframes = RPitch->size[1] - 1;
  519     numcands = RPitch->size[0] - 1;
  520   
  521     /* Copy some arrays */
  522     ix = RPitch->size[1] - 1;
  523     n = 0;
  524     emxFree_real_T(&b_DataD);
  525     emxFree_real_T(&y);
  526     emxFree_real_T(&TMerit2);
  527     emxFree_real_T(&TPitch2);
  528     emxFree_real_T(&TMerit1);
  529     emxFree_boolean_T(&VUVEnergy);
  530     emxFree_real_T(&DataD);
  531     emxFree_real_T(&DataB);
  532     for (loop_ub = 0; loop_ub <= ix; loop_ub++) {
  533       if (RPitch->data[(numcands + RPitch->size[0] * loop_ub) - 1] > 0.0) {
  534         n++;
  535       }
  536     }
  537   
  538     i0 = r0->size[0] * r0->size[1];
  539     r0->size[0] = 1;
  540     r0->size[1] = n;
  541     emxEnsureCapacity((emxArray__common *)r0, i0, (int)sizeof(int));
  542     n = 0;
  543     for (loop_ub = 0; loop_ub <= ix; loop_ub++) {
  544       if (RPitch->data[(numcands + RPitch->size[0] * loop_ub) - 1] > 0.0) {
  545         r0->data[n] = loop_ub + 1;
  546         n++;
  547       }
  548     }
  549   
  550     emxInit_real_T(&b_RPitch, 2);
  551     emxInit_real_T(&c_RPitch, 2);
  552     loop_ub = RPitch->size[1];
  553     n = RPitch->size[0];
  554     i0 = c_RPitch->size[0] * c_RPitch->size[1];
  555     c_RPitch->size[0] = 1;
  556     c_RPitch->size[1] = loop_ub;
  557     emxEnsureCapacity((emxArray__common *)c_RPitch, i0, (int)sizeof(double));
  558     for (i0 = 0; i0 < loop_ub; i0++) {
  559       c_RPitch->data[c_RPitch->size[0] * i0] = RPitch->data[(n + RPitch->size[0] *
  560         i0) - 2];
  561     }
  562   
  563     i0 = b_RPitch->size[0] * b_RPitch->size[1];
  564     b_RPitch->size[0] = 1;
  565     b_RPitch->size[1] = r0->size[1];
  566     emxEnsureCapacity((emxArray__common *)b_RPitch, i0, (int)sizeof(double));
  567     loop_ub = r0->size[1];
  568     for (i0 = 0; i0 < loop_ub; i0++) {
  569       b_RPitch->data[b_RPitch->size[0] * i0] = c_RPitch->data[r0->data[r0->size[0]
  570         * i0] - 1];
  571     }
  572   
  573     emxFree_real_T(&c_RPitch);
  574     emxFree_int32_T(&r0);
  575     emxInit_real_T1(&Trans, 3);
  576     mean_pitch = mean(b_RPitch);
  577   
  578     /* The following weighting factors are used to differentially weight */
  579     /*  the various types of transitions which can occur, as well as weight */
  580     /*  the relative value of transition costs and local costs */
  581     /* -- MAIN ROUTINE -------------------------------------------------------------- */
  582     /*  Forming the local cost matrix */
  583     /*  Initialization for the formation of the transition cost matrix */
  584     i0 = Trans->size[0] * Trans->size[1] * Trans->size[2];
  585     Trans->size[0] = RPitch->size[0];
  586     Trans->size[1] = RPitch->size[0];
  587     Trans->size[2] = RPitch->size[1];
  588     emxEnsureCapacity((emxArray__common *)Trans, i0, (int)sizeof(double));
  589     loop_ub = RPitch->size[0] * RPitch->size[0] * RPitch->size[1];
  590     emxFree_real_T(&b_RPitch);
  591     for (i0 = 0; i0 < loop_ub; i0++) {
  592       Trans->data[i0] = 1.0;
  593     }
  594   
  595     /*  The transition cost matrix is proportional to frequency differences */
  596     /*  between successive candidates. */
  597     for (loop_ub = 1; loop_ub - 1 < numframes; loop_ub++) {
  598       for (j = 0; j <= numcands; j++) {
  599         for (k = 0; k <= numcands; k++) {
  600           /*  both candidates voiced */
  601           if ((RPitch->data[j + RPitch->size[0] * loop_ub] > 0.0) && (RPitch->
  602                data[k + RPitch->size[0] * (loop_ub - 1)] > 0.0)) {
  603             Trans->data[(k + Trans->size[0] * j) + Trans->size[0] * Trans->size[1]
  604               * loop_ub] = Parameter[28] * (fabs(RPitch->data[j + RPitch->size[0] *
  605               loop_ub] - RPitch->data[k + RPitch->size[0] * (loop_ub - 1)]) /
  606               mean_pitch);
  607           }
  608   
  609           /*  one candiate is unvoiced */
  610           if (((RPitch->data[j + RPitch->size[0] * loop_ub] == 0.0) &&
  611                (RPitch->data[k + RPitch->size[0] * (loop_ub - 1)] > 0.0)) ||
  612               ((RPitch->data[j + RPitch->size[0] * loop_ub] > 0.0) &&
  613                (RPitch->data[k + RPitch->size[0] * (loop_ub - 1)] == 0.0))) {
  614             s = fabs(Energy->data[loop_ub - 1] - Energy->data[loop_ub]);
  615             if ((1.0 <= s) || rtIsNaN(s)) {
  616               d0 = 1.0;
  617             } else {
  618               d0 = s;
  619             }
  620   
  621             Trans->data[(k + Trans->size[0] * j) + Trans->size[0] * Trans->size[1]
  622               * loop_ub] = Parameter[29] * (1.0 - d0);
  623           }
  624   
  625           /*  both candidates are unvoiced */
  626           if ((RPitch->data[j + RPitch->size[0] * loop_ub] == 0.0) &&
  627               (RPitch->data[k + RPitch->size[0] * (loop_ub - 1)] == 0.0)) {
  628             Trans->data[(k + Trans->size[0] * j) + Trans->size[0] * Trans->size[1]
  629               * loop_ub] = Parameter[30];
  630           }
  631         }
  632       }
  633     }
  634   
  635     /*  Overal balance between Local and Transition costs */
  636     i0 = Trans->size[0] * Trans->size[1] * Trans->size[2];
  637     emxEnsureCapacity((emxArray__common *)Trans, i0, (int)sizeof(double));
  638     n = Trans->size[0];
  639     ix = Trans->size[1];
  640     iy = Trans->size[2];
  641     loop_ub = n * ix * iy;
  642     for (i0 = 0; i0 < loop_ub; i0++) {
  643       Trans->data[i0] /= Parameter[31];
  644     }
  645   
  646     emxInit_real_T(&r1, 2);
  647   
  648     /*  Find the minimum cost path thru Pitch_Array using the Local and Trans costs */
  649     i0 = r1->size[0] * r1->size[1];
  650     r1->size[0] = Merit->size[0];
  651     r1->size[1] = Merit->size[1];
  652     emxEnsureCapacity((emxArray__common *)r1, i0, (int)sizeof(double));
  653     loop_ub = Merit->size[0] * Merit->size[1];
  654     for (i0 = 0; i0 < loop_ub; i0++) {
  655       r1->data[i0] = 1.0 - Merit->data[i0];
  656     }
  657   
  658     emxFree_real_T(&Merit);
  659     path1(r1, Trans, Path_data, Path_size);
  660   
  661     /* extracting the pitch, using Path */
  662     i0 = Data_temp->size[0] * Data_temp->size[1];
  663     Data_temp->size[0] = 1;
  664     Data_temp->size[1] = RPitch->size[1];
  665     emxEnsureCapacity((emxArray__common *)Data_temp, i0, (int)sizeof(double));
  666     loop_ub = RPitch->size[1];
  667     emxFree_real_T(&r1);
  668     emxFree_real_T(&Trans);
  669     for (i0 = 0; i0 < loop_ub; i0++) {
  670       Data_temp->data[i0] = 0.0;
  671     }
  672   
  673     for (loop_ub = 0; loop_ub <= numframes; loop_ub++) {
  674       Data_temp->data[loop_ub] = RPitch->data[((int)Path_data[loop_ub] +
  675         RPitch->size[0] * loop_ub) - 1];
  676     }
  677   
  678     emxFree_real_T(&RPitch);
  679     *numfrms = Data_temp->size[1];
  680     *frmrate = Parameter[1];
  681     freqSelect(Data_temp, Energy, DataC);
  682   
  683     /*  Preprocess the raw data from android voice recorder */
  684     /*  */
  685     /*  Data_in = Raw data android voice recorder */
  686     /*  Data_out = Head and tail muted data */
  687     /*  The noise comes from opening mic and closing mic */
  688     i0 = Pitch->size[0] * Pitch->size[1];
  689     Pitch->size[0] = 1;
  690     Pitch->size[1] = Energy->size[1];
  691     emxEnsureCapacity((emxArray__common *)Pitch, i0, (int)sizeof(double));
  692     loop_ub = Energy->size[0] * Energy->size[1];
  693     emxFree_real_T(&Data_temp);
  694     emxFree_real_T(&DataC);
  695     for (i0 = 0; i0 < loop_ub; i0++) {
  696       Pitch->data[i0] = Energy->data[i0];
  697     }
  698   
  699     for (loop_ub = 0; loop_ub < Energy->size[1]; loop_ub++) {
  700       if ((loop_ub + 2U < (unsigned int)Pitch->size[1]) && (loop_ub > 0) &&
  701           (Pitch->data[loop_ub - 1] != Pitch->data[loop_ub]) && (Pitch->
  702            data[loop_ub] != Pitch->data[loop_ub + 1])) {
  703         Pitch->data[loop_ub] = Pitch->data[loop_ub - 1];
  704       }
  705     }
  706   
  707     emxFree_real_T(&Energy);
  708     Smooth(Pitch, Parameter[32]);
  709     Smooth(Pitch, Parameter[32]);
  710     Smooth(Pitch, Parameter[32]);
  711

  717     /*  hold off */
  718     /*  title ('spectral pitch track') */
  719     /*  pause */
  720     /*   If VU = 0,  be sure entire track is voiced */
  721     /*      if (VU == 0)      %  Only applies if we want all voiced track     */
  722     /*         IDX_unvoiced = (Pitch < 1); */
  723     /*           Num_UV       = sum (IDX_unvoiced);    % Total number unvoiced frames */
  724     /*            if (Num_UV > 0)  */
  725     /*              for n = 1:numfrms;    %  there should be a more efficient way to do this */
  726     /*                 if (IDX_unvoiced(n) == 1); */
  727     /*                   Pitch(n) = SPitch(n); */
  728     /*                  end;   */
  729     /*              end;  */
  730     /*            end;   */
  731     /*       end; */

  733     /*  if (VU==0) */
  734     /*      Pitch = ptch_fix(Pitch); */
  735     /*  end */
