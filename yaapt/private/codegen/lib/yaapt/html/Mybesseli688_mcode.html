1function output = Mybesseli(alpha,xx,scale)
  2%BESSELI Modified Bessel functions of the first kind.
  3%   I = BESSELI(ALPHA,X) computes modified Bessel functions of the
  4%   first kind, I_sub_alpha(X) for real, non-negative order ALPHA
  5%   and argument X.  In general, both ALPHA and X may be vectors.
  6%   The output I is m-by-n with m = length(X), n = length(ALPHA) and
  7%       I(k,j) = I_sub_alpha(j)(X(k)).
  8%   The elements of X can be any nonnegative real values in any order.
  9%   For ALPHA there are two important restrictions: the increment
 10%   in ALPHA must be one, i.e. ALPHA = alpha:1:alpha+n-1, and the
 11%   elements must also be in the range 0 <= ALPHA(j) <= 1000.
 12%
 13%   E = BESSELI(ALPHA,X,1) computes I_sub_alpha(X)*EXP(-X).
 14%
 15%   The relationship between the Bessel functions of the first kind is
 16%
 17%       I_sub_alpha(x) = i^(-alpha) * J_sub_alpha(i*x)
 18%
 19%   Examples:
 20%
 21%       besseli(3:9,[0:.2:9.8 10:.5:20],1) generates the entire 
 22%       71-by-7 table on page 423 of Abramowitz and Stegun,
 23%       "Handbook of Mathematical Functions."
 24%
 25%   See also: BESSELJ, BESSELY, BESSELK.
 26
 27%   Acknowledgement:
 28%
 29%   This program is based on a Fortran program by W. J. Cody and
 30%   L. Stoltz, Applied Mathematics Division, Argonne National
 31%   Laboratory, dated May 30, 1989.  Their references include:
 32%
 33%      "A Note on backward recurrence algorithms," Olver, F. W. J.,
 34%      and Sookne, D. J., Math. Comp. 26, 1972, pp 941-947.
 35%
 36%      "Bessel functions of real argument and integer order,"
 37%      Sookne, D. J., NBS Jour. of Res. B. 77b, 1973, pp. 125-132.
 38%
 39%      "Algorithm 597, Sequence of modified Bessel functions of the
 40%      "first kind," Cody, W. J., Trans. Math. Soft., 1983, pp. 242-245.
 41%
 42%   MATLAB version: C. Moler, 10/6/92.
 43%   Copyright (c) 1984-94 by The MathWorks, Inc.
 44
 45%
 46%  Check for real, non-negative arguments.
 47%
 48%    if nargin < 3, scale = 0; end
 49%    if any(imag(xx)) | any(xx < 0) | any(imag(alpha)) | any(alpha < 0)
 50%       error('Input arguments must be real and nonnegative.')
 51%    end
 52%    if isempty(alpha) | isempty(xx)
 53%       bk = []; 
 54%       return
 55%    end
 56   scale = 0;
 57%
 58%  Break alpha into integer and fractional parts,
 59%  and initialize result array.
 60%
 61   nfirst = fix(alpha(1));
 62   nb = fix(alpha(length(alpha))) + 1;
 63   b = NaN*ones(length(xx),nb);
 64   if ~(nb <= 1001)
 65      error('Alpha must be <= 1000.')
 66   end
 67   if length(alpha) > 1
 68      if any(abs(diff(alpha)-1) > 4*nb*eps)
 69         error('Increment in alpha must be 1.')
 70      end
 71   end
 72%    resize = (length(alpha) == 1);
 73   resize_flag = (length(alpha) == 1);
 74%    if resize, resize = size(xx); end
 75   if resize_flag
 76       resize = size(xx); 
 77   end
 78   xx = xx(:);
 79%    b = NaN*ones(length(xx),nb);
 80   alpha = alpha(1) - nfirst;
 81%
 82%  Asymptotic expansion for large x.
 83%
 84   v = find(1.e3 < xx);
 85   if any(v)
 86      x = xx(v);
 87      if scale
 88         z = 1./(8*x);
 89         mu = 4*(nfirst+alpha:1:nb-1+alpha).^2;
 90         z = (1 - z*(mu-1).*(1 - (z/2)*(mu-9).*(1 - (z/3)*(mu - 25))));
 91         b(v,nfirst+1:nb) = ((1./sqrt(2*pi*x))*ones(1,nb-nfirst)).*z;
 92      else
 93         b(v,:) = Inf*ones(length(v),nb);
 94      end
 95   end
 96%
 97%  Three-term backward recurrence for most x.
 98%
 99   v = find((1.e-4 <= xx) & (xx <= 1.e3));
100   if any(v)
101      x = xx(v);
102      magx = max(fix(x));
103%
104%     Initialize the forward sweep, the p-sequence of Olver.
105%
106      nbmx = nb-magx;
107      n = magx+1;
108      en = (n+n) + (alpha+alpha);
109      plast = 1;
110      p = en ./ x;
111%
112%     Calculate general significance test.
113%
114      test = 2/eps;
115      if (2*magx > 80);
116         test = sqrt(test*max(p));
117      else
118         test = test / 1.585^magx;
119      end
120%       if (nbmx >= 3)
121% %
122% %     Calculate p-sequence until n = nb-1.
123% %
124%          tover = eps*realmax;
125%          nstart = magx+2;
126%          nend = nb - 1;
127%          for k = nstart:nend
128%             n = k;
129%             en = en + 2;
130%             pold = plast;
131%             plast = p;
132%             p = en * plast./x + pold;
133%             if any(p > tover)
134% %
135% %           To avoid overflow, divide p-sequence by tover.
136% %           Calculate p-sequence until abs(p) > 1.
137% %
138%                tover = realmax;
139%                p = p / tover;
140%                plast = plast / tover;
141%                psave = p;
142%                psavel = plast;
143%                nstart = n + 1;
144%                while any(p <= 1);
145%                   n = n + 1;
146%                   en = en + 2;
147%                   pold = plast;
148%                   plast = p;
149%                   p = en * plast./x + pold;
150%                end
151%                tempb = en ./ x;
152% %
153% %              Calculate backward test, and find ncalc,
154% %              the highest n such that the test is passed.
155% %
156%                test = pold*plast*eps;
157%                test = test*(0.5-0.5/min(tempb*tempb));
158%                p = plast * tover;
159%                n = n - 1;
160%                en = en - 2;
161%                nend = min(nb,n);
162%                ncalc = nend + 1;
163%                for l = nstart:nend
164%                   pold = psavel;
165%                   psavel = psave;
166%                   psave = en * psavel./x + pold;
167%                   if any(psave.*psavel > test);
168%                      ncalc = l-1;
169%                      test = 0;
170%                      break
171%                   end
172%                end
173%                break
174%             end
175%          end
176%          if ~test
177%             n = nend;
178%             en = (n+n) + (alpha+alpha);
179% %
180% %           Calculate special significance test for nbmx > 2.
181% %
182%             test = max(test,max(sqrt(plast/eps).*sqrt(p+p)));
183%          end
184%       end
185%
186%     Calculate p-sequence until significance test passed.
187%
188      a = p;
189      count = 0;
190      while any(p < test)
191         n = n + 1;
192         en = en + 2;
193         if count == 1
194            plast = a;
195         else
196             count = 1;
197         end
198         pold = plast;
199         a = p;
200         p = en * a./x + pold;
201      end
202%
203%     Initialize the backward recursion and the normalization sum.
204%
205      n = n + 1;
206      en = en + 2;
207      tempb = 0;
208      tempa = 1 ./ p;
209      em = n - 1;
210      empal = em + alpha;
211      emp2al = (em - 1) + (alpha + alpha);
212      sum = tempa * empal * emp2al / em;
213      nend = n - nb;
214      skip = 0;
215      if (nend < 0)
216         b(v,n) = tempa;
217         nend = -nend;
218         b(v,n+1:n+nend) = zeros(length(v),nend);
219      else
220         if (nend > 0)
221%
222%           Recur backward via difference equation, calculating 
223%           (but not storing) b(n), until n = nb.
224%
225            kk= tempa;
226            count2 = 0;
227            for l = 1:nend
228               n = n - 1;
229               en = en - 2;
230               if count2 == 1
231                   tempb = kk;
232               else
233                   count2 = 1;
234               end
235               tempc = tempb;
236               kk = tempa;
237               tempa = (en*kk) ./ x + tempc;
238               em = em - 1;
239               emp2al = emp2al - 1;
240               if (n == 1), break, end
241               if (n == 2), emp2al = 1; end
242               empal = empal - 1;
243               sum = (sum + tempa*empal) * emp2al / em;
244            end
245         end
246%
247%        Store b(nb).
248%
249         b(v,n) = tempa;
250         if (nb <= 1)
251            sum = (sum + sum) + tempa;
252            skip = -1;
253         else
254%
255%           Calculate and store b(nb-1).
256%
257            n = n - 1;
258            en = en - 2;
259            b(v,n)  = (en*tempa) ./ x + tempb;
260            if (n == 1)
261               skip = 1;
262            else
263               em = em - 1;
264               emp2al = emp2al - 1;
265               if (n == 2), emp2al = 1; end
266               empal = empal - 1;
267               sum = (sum + b(v,n)*empal) * emp2al / em;
268            end
269         end
270      end
271      if skip == 0
272         nend = n - 2;
273         if (nend > 0)
274%
275%        Calculate via difference equation and store b(n), until n = 2.
276%
277            for l = 1:nend
278               n = n - 1;
279               en = en - 2;
280               b(v,n) = (en*b(v,n+1)) ./ x + b(v,n+2);
281               em = em - 1;
282               emp2al = emp2al - 1;
283               if (n == 2), emp2al = 1; end
284               empal = empal - 1;
285               sum = (sum + b(v,n)*empal) * emp2al / em;
286            end
287         end
288      end
289%
290%     Calculate b(1)
291%
292%       if skip == 0
293%           b(v,1) = 2*empal*b(v,2) ./ x + b(v,3); 
294%       end
295%       if skip >= 0
296%           sum = (sum + sum) + b(v,1); 
297%       end
298%
299%     Normalize.  Divide all b(n) by sum.
300%
301      if (alpha ~= 0)
302         sum = sum * gamma(1+alpha) .* (x*0.5).^(-alpha);
303      end
304      if ~scale, sum = sum .* exp(-x); end
305      for n = 1:nb
306         b(v,n) = b(v,n) ./ sum;
307      end
308   end
309%
310%  Two-term ascending series for small x.
311%
312   v = find((0 < xx) & (xx < 1.e-4));
313   if any(v)
314      x = xx(v);
315      tempa = ones(length(x),1);
316      empal = 1 + alpha;
317      halfx = 0.5 * x;
318      if (alpha ~= 0),  tempa = halfx.^alpha /gamma(empal); end
319      if scale, tempa = tempa .* exp(-x); end
320      tempb = halfx .* halfx;
321      b(v,1) = tempa + tempa.*tempb / empal;
322      if (nb > 1)
323%
324%        Calculate higher-order functions.
325%
326         tempc = halfx;
327         tover = 8*realmin ./ x;
328         for n = 2:nb;
329            tempa = tempa / empal;
330            empal = empal + 1;
331            tempa = tempa .* tempc;
332            tempa = tempa .* (tempa > tover*empal);
333            b(v,n) = tempa + tempa.*tempb / empal;
334         end
335      end
336   end
337%
338%  x == 0
339%
340   v = find(xx == 0);
341   if any(v)
342      % if alpha>0, I(alpha,0) = 0; I(0,0) = 1.
343      b(v,:) = zeros(length(v),nb);
344      if alpha == 0, b(v,1) = ones(length(v),1); end
345   end
346%
347%  Return the requested index range
348%
349   b = b(:,nfirst+1:nb);
350%    if resize
351   if resize_flag
352      output = reshape(b,resize(1),resize(2));
353   else
354       output = b;
355   end
356end
